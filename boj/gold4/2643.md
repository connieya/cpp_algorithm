# BOJ 2643 색종이 올려 놓기

## solution 1

```c
#include "bits/stdc++.h"

using namespace std;

vector<pair<int, int>> v;
int dp[101];

bool comp(pair<int, int> a, pair<int, int> b) {
    if (a.first == b.first) {
        return a.second < b.second;
    }
    return a.first < b.first;
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n, a, b;
    cin >> n;
    for (int i = 0; i < n; ++i) {
        cin >> a >> b;
        if (b > a) swap(a, b);
        v.push_back({a, b});
    }
    sort(v.begin(), v.end(), comp);
    dp[0] = 1;
    int ans = 1;
    for (int i = 1; i < n; ++i) {
        dp[i] = 1;
        for (int j = 0; j < i; ++j) {
            if (v[i].first >= v[j].first && v[i].second >= v[j].second && dp[i] < dp[j] + 1) {
                dp[i] = dp[j] + 1;
            }
        }
        ans = max(ans, dp[i]);
    }
    cout << ans;

}
```
