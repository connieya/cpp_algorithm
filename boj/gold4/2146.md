## solution 1

```c
#include "stdio.h"
#include "queue"
#include "vector"
#include "algorithm"
using namespace std;

int map[100][100];
bool visited[100][100];
int n;
int dx[] = {-1, 0, 1, 0};
int dy[] = {0, 1, 0, -1};
vector<pair<int, int >> v;

void bfs(int x, int y, int num);

int main(void) {
    int num = 1;
    int ans = 214700;
    scanf("%d", &n);
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            scanf("%d", &map[i][j]);
        }
    }
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            if (map[i][j] && !visited[i][j]) {
                map[i][j] = num;
                visited[i][j] = true;
                bfs(i, j, num);
                num++;
            }
        }
    }
    int tmp = 0;
    for (int i = 0; i < v.size() - 1; ++i) {
        int x1 = v[i].first;
        int y1 = v[i].second;
        for (int j = i + 1; j < v.size(); ++j) {
            int x2 = v[j].first;
            int y2 = v[j].second;
            if (map[x1][y1] != map[x2][y2]) {
                tmp = abs(x1 - x2) + abs(y1 - y2) - 1;
                ans = min(ans, tmp);
            }
        }
    }
    printf("%d",ans);
}

void bfs(int x, int y, int num) {
    queue<pair<int, int >> Q;
    Q.push({x, y});
    while (!Q.empty()) {
        bool isEdge = false;
        auto cur = Q.front();
        Q.pop();
        for (int i = 0; i < 4; ++i) {
            int nx = cur.first + dx[i];
            int ny = cur.second + dy[i];
            if (nx < 0 || nx >= n || ny < 0 || ny >= n) continue;
            if (map[nx][ny] && !visited[nx][ny]) {
                visited[nx][ny] = true;
                map[nx][ny] = num;
                Q.push({nx, ny});
            }
            if (!map[nx][ny]) {
                isEdge = true;
            }
        }
        if (isEdge){
            v.push_back({cur.first, cur.second});
        }
    }
}
```

## solution 2

### 시간  32ms => 12ms

````c
#include "bits/stdc++.h"

using namespace std;

int n , ans = INT_MAX;
int dx[] = {-1, 0, 1, 0};
int dy[] = {0, 1, 0, -1};
int mp[101][101];
int dist[101][101];
bool visited[101][101];
vector<pair<int, int>> v;

void bfs(int x, int y, int idx) {
    visited[x][y] = true;
    mp[x][y] = idx;
    queue<pair<int, int>> q;
    q.push({x, y});
    while (!q.empty()) {
        int x = q.front().first;
        int y = q.front().second;
        q.pop();
        for (int i = 0; i < 4; ++i) {
            int nx = x + dx[i];
            int ny = y + dy[i];
            if (nx < 0 || nx >= n || ny < 0 || ny >= n || !mp[nx][ny]) continue;
            if (!visited[nx][ny]) {
                q.push({nx, ny});
                visited[nx][ny] = true;
                mp[nx][ny] = idx;
            }
        }
    }
}

void dfs(int x, int y) {
    for (int i = 0; i < 4; ++i) {
        int nx = x + dx[i];
        int ny = y + dy[i];
        if (nx < 0 || nx >= n || ny < 0 || ny >= n) continue;
        if (!visited[nx][ny]) {
            visited[nx][ny] = true;
            if (mp[nx][ny]) {
                v.push_back({nx, ny});
                continue;
            }
            dfs(nx, ny);
        }
    }
}

void bridge(int x, int y) {
    dist[x][y] = 0;
    int value = mp[x][y];
    queue<pair<int, int>> q;
    q.push({x, y});
    while (!q.empty()) {
        int x = q.front().first;
        int y = q.front().second;
        q.pop();
        for (int i = 0; i < 4; ++i) {
            int nx = x + dx[i];
            int ny = y + dy[i];
            if (nx < 0 || nx >= n || ny < 0 || ny >= n || mp[nx][ny] == value) continue;
            if(mp[nx][ny] && mp[nx][ny] != value){
                if(ans > dist[x][y]){
                    ans = dist[x][y];
                }
                continue;
            }
            if(dist[nx][ny] == -1 || dist[nx][ny] > dist[x][y]+1){
                dist[nx][ny] = dist[x][y]+1;
                q.push({nx,ny});
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    cin >> n;
    int x, y;
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            cin >> mp[i][j];
            dist[i][j] = -1;
            if (mp[i][j] == 0) {
                x = i, y = j;
            }
        }
    }
    int idx = 1;
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            if (mp[i][j] && !visited[i][j]) {
                bfs(i, j, idx);
                idx++;
            }
        }
    }
    memset(visited, false, sizeof(visited));
    dfs(x, y);
    for (int i = 0; i < v.size(); ++i) {
        bridge(v[i].first,v[i].second);
    }
    cout << ans << '\n';
}
````
