# BOJ 16637 괄호 추가하기

## solution 1

### 비트 마스킹

```c
#include "bits/stdc++.h"

using namespace std;

vector<char> op;
vector<int> num;

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n;
    char ch;
    cin >> n;
    for (int i = 0; i < n; ++i) {
        cin >> ch;
        if(i % 2){
            op.push_back(ch);
        }else {
            num.push_back(ch-'0');
        }
    }

    int m  = (n-1)/2;
    int ans = INT_MIN;
    for (int i = 0; i < 1<<m; ++i) {
        bool flag = false;
        for (int j = 0; j < m-1; ++j) {
            if(i & (1<<j) && i & (1<<(j+1))) flag = true;
        }

        if(flag) continue;
        vector<int> tmp = num;
        vector<char> oop = op;
        for (int j = 0; j < m; ++j) {
            if(i & (1<<j)){
                if(oop[j] == '+'){
                    tmp[j] += tmp[j+1];
                    tmp[j+1] = 0;
                }else if(oop[j] == '*'){
                    tmp[j] *= tmp[j+1];
                    oop[j] = '+';
                    tmp[j+1] = 0;
                }else {
                    tmp[j] -= tmp[j+1];
                    oop[j] = '+';
                    tmp[j+1] = 0;
                }
            }
        }
        int res = tmp[0];
        for (int j = 0; j < m; ++j) {
            if(oop[j] == '+'){
               res += tmp[j+1];
            }else if(oop[j] == '*'){
                res *= tmp[j+1];
            }else {
                res -= tmp[j+1];
            }
        }
        if(res > ans){
            ans = res;
        }
    }
    cout << ans;
}
```

## solution 2

### DFS


```c
#include "bits/stdc++.h"

using namespace std;

vector<char> op;
vector<int> num;
int n;
int ans = INT_MIN;

int calc(int a, int b, char op) {
    if (op == '+') return a + b;
    if (op == '*') return a * b;
    return a - b;
}

void dfs(int L, int sum) {
    if (L >= n) {
        if (L == n && sum > ans) {
            ans = sum;
        }
        return;
    }
    dfs(L + 1, calc(sum, num[L + 1], op[L]));
    dfs(L + 2, calc(sum, calc(num[L + 1], num[L + 2], op[L + 1]), op[L]));
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    char ch;
    cin >> n;
    for (int i = 0; i < n; ++i) {
        cin >> ch;
        if (i % 2) {
            op.push_back(ch);
        } else {
            num.push_back(ch - '0');
        }
    }

    n /= 2;
    dfs(0, num[0]);
    cout << ans;
}
```
