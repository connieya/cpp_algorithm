# BOJ 17406 배열 돌리기 4

## Solution 1


```c
#include "bits/stdc++.h"

using namespace std;

int n, m, k;
int board[51][51];
int tmp[51][51];
vector<tuple<int, int, int>> v;

void rotate(int x1, int y1, int x2, int y2) {
    int tmp = board[x2][y1];
    for (int i = y1; i < y2; ++i) {
        board[x2][i] = board[x2][i + 1];
    }
    for (int i = x2; i > x1; --i) {
        board[i][y2] = board[i - 1][y2];
    }
    for (int i = y2; i > y1; --i) {
        board[x1][i] = board[x1][i - 1];
    }
    for (int i = x1; i < x2; ++i) {
        board[i][y1] = board[i + 1][y1];
    }
    board[x2 - 1][y1] = tmp;
}

void rotate(int r, int c, int s) {
    int x1 = r - s;
    int y1 = c - s;
    int x2 = r + s;
    int y2 = c + s;
    while (x1 < x2 && y1 < y2) {
        rotate(x1++, y1++, x2--, y2--);
    }
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    cin >> n >> m >> k;

    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            cin >> tmp[i][j];
        }
    }
    for (int i = 0; i < k; ++i) {
        int a, b, c;
        cin >> a >> b >> c;
        v.push_back({a - 1, b - 1, c});
    }
    vector<int> perm;
    for (int i = 0; i < k; ++i) {
        perm.push_back(i);
    }
    int ans = INT_MAX;
    int r,c,s;
    do {
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                board[i][j] = tmp[i][j];
            }
        }
        for (auto idx: perm) {
            tie(r, c, s) = v[idx];
            rotate(r, c, s);
        }
        int tp = INT_MAX;
        for (int i = 0; i < n; ++i) {
            int sum = 0;
            for (int j = 0; j < m; ++j) {
                sum += board[i][j];
            }
            if (tp > sum) tp = sum;
        }
        if (ans > tp) ans = tp;
    } while (next_permutation(perm.begin(), perm.end()));
    cout << ans;
}
```
