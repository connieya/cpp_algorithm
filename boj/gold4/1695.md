# BOJ 1695 팰린드롬 만들기

## solution 1
```c
#include "bits/stdc++.h"

using namespace std;

int n;
int arr[5001];
int cache[5001][5001];

int solve(int s, int e) {
    if (s > e) return 0;
    int &ret = cache[s][e];
    if (ret != -1) return ret;
    if (arr[s] == arr[e]) return ret = solve(s + 1, e - 1);
    return ret = min(solve(s + 1, e), solve(s, e - 1)) + 1;
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    cin >> n;
    for (int i = 1; i <= n; ++i) {
        cin >> arr[i];
    }
    memset(cache, -1, sizeof(cache));
    cout << solve(1, n);
}
```
