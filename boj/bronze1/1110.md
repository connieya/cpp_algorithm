# BOJ 1110 더하기 사이클

## solution 1

```c
#include "bits/stdc++.h"

using namespace std;

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n;
    cin >> n;
    int answer = 0;
    int target = n;
    while (true) {
        target = (target % 10) * 10 + ((target / 10 + target % 10)) % 10;
        answer++;
        if(target == n) break;
    }
    cout << answer;
}
```

## solution 2

```c
#include "bits/stdc++.h"

using namespace std;

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n;
    cin >> n;
    int answer = 0;
    int target = n;
    do {
        target = (target % 10) * 10 + ((target / 10 + target % 10)) % 10;
        answer++;
    } while (target != n);
    cout << answer;
}
```
