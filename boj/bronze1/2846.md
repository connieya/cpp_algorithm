# BOJ 2846 오르막길

## solution 1 구현

```c++
#include "bits/stdc++.h"

using namespace std;


int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    int n;
    cin >> n;
    int arr[10001];
    for(int i=0; i<n; i++){
        cin >> arr[i];
    }
    int ans = 0;
    int prev = arr[0];
    for(int i=1; i<n; i++) {
        if(arr[i] >  arr[i-1]){
            ans = max(ans, arr[i]-prev);
            continue;
        }
        prev = arr[i];
    }
    cout << ans << '\n';


}
```
