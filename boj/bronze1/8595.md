# BOJ 8595 히든 넘버

### 문제 꼼꼼히 읽기, 범위 신경 쓰기 (오버 플로우 조심)

## Solution 1

```c++
#include "bits/stdc++.h"

using namespace std;

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    long long n, ans = 0;
    string str;
    cin >> n;
    cin >> str;
    for(char &s: str) {
        if (isalpha(s)){
            s = ' ';
        }
    }
    stringstream iss(str);
    while (iss >> str){
        ans += stoi(str);
    }
    cout << ans;

}
```

## Solution 2

```c++
#include "bits/stdc++.h"

using namespace std;

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    long long n, ans = 0;
    string str;
    cin >> n;
    cin >> str;
    for (int i = 0; i < n;) {
        string tmp = "";
        while (i < n && isdigit(str[i])) {
            tmp += str[i++];
        }
        while (i < n && !isdigit(str[i])) {
            i++;
        }
        if (tmp.length() > 0) {
            ans += stoi(tmp);
        }
    }
    cout << ans;
}
```

## solution 3

```c
#include "bits/stdc++.h"

using namespace std;

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    long long ans = 0;
    long long res = 0;
    int n;
    string str;
    cin >> n >> str;
    for (char ch: str) {
        if (isdigit(ch)) {
            res = res * 10 + (ch - '0');
        } else {
            ans += res;
            res = 0;
        }
    }
    cout << ans+res;
}

```
