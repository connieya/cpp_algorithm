# BOJ 9376 탈옥

## solution 1

```c++
#include "bits/stdc++.h"

using namespace std;

int dx[] = {-1, 0, 1, 0};
int dy[] = {0, 1, 0, -1};

vector<vector<int>> bfs(vector<string> &a, int x, int y) {
    int n = a.size();
    int m = a[0].size();
    vector<vector<int>> d(n, vector<int>(m));
    for (int i=0; i<n; i++) {
        for (int j=0; j<m; j++) {
            d[i][j] = -1;
        }
    }

    deque<pair<int, int>> q;
    q.push_back({x, y});
    d[x][y] = 0;
    while (!q.empty()) {
        int x = q.front().first;
        int y = q.front().second;
        q.pop_front();
        for (int i = 0; i < 4; ++i) {
            int nx = x + dx[i];
            int ny = y + dy[i];
            if (nx < 0 || nx >= n || ny < 0 || ny >= m) continue;
            if (a[nx][ny] == '*' || d[nx][ny] != -1) continue;
            if (a[nx][ny] == '#') {
                q.push_back({nx, ny});
                d[nx][ny] = d[x][y] + 1;
            } else {
                q.push_front({nx, ny});
                d[nx][ny] = d[x][y];
            }
        }
    }
    return d;
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int t;
    cin >> t;
    while (t--) {
        int n,m;
        cin >> n >> m;
        int res = 1e8;
        vector<string> a(n + 2);
        for (int i = 1; i <= n; ++i) {
            cin >> a[i];
            a[i] = "." + a[i] + ".";
        }
        n += 2;
        m += 2;
        for (int i = 0; i < m; ++i) {
            a[0] += '.';
            a[n - 1] += '.';
        }
        vector<vector<int>> d1 = bfs(a, 0, 0);
        int x1, y1, x2, y2;
        x1 = x2 = y1 = y2 = -1;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (a[i][j] == '$') {
                    if (x1 == -1) {
                        x1 = i;
                        y1 = j;
                    } else {
                        x2 = i;
                        y2 = j;
                    }
                }
            }
        }
        vector<vector<int>> d2 = bfs(a, x1, y1);
        vector<vector<int>> d3 = bfs(a, x2, y2);

        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (a[i][j] == '*') continue;
                if (d1[i][j] == -1 || d2[i][j] == -1 || d3[i][j] == -1) continue;
                int ans = d1[i][j] + d2[i][j] + d3[i][j];
                if (a[i][j] == '#') ans -= 2;
                if (res > ans) {
                    res = ans;
                }

            }
        }
        cout << res << '\n';
    }
}
```
