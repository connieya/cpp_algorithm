# BOJ 10162 전자레인지

## solution 1

```c
#include "bits/stdc++.h"

using namespace std;

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int t ;
    int arr[] = {300,60,10};
    int ans[3] ={};
    cin >> t;
    for (int i = 0; t && i < 3; ++i) {
        if(t >= arr[i]){
            ans[i] += t/arr[i];
            t %= arr[i];
        }
    }
    if(t){
        cout << -1;
        exit(0);
    }
    cout << ans[0] << ' ' << ans[1] << ' ' << ans[2];
}
```

## solution 2

```c
#include "bits/stdc++.h"

using namespace std;

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int t;
    cin >> t;
    if (t % 10) {
        cout << -1;
    } else {
        cout << t / 300 << ' ' << t % 300 / 60 << ' ' << t % 300 % 60 / 10;
    }
}
```
