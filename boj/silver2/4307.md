# BOJ 4307 개미

## Solution 1

### 시간 40 ms

max 함수를 사용함

```c
#include "bits/stdc++.h"

using namespace std;

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int t;
    cin >> t;
    while (t--) {
        int stick , n, mx = -1, mn = -1;
        cin >> stick >> n;
        for (int i = 0; i < n; ++i) {
            int d;
            cin >> d;
           int t_mn = min(stick-d ,d);
           int t_mx = max(stick-d,d);
           mn = max(mn,t_mn);
           mx = max(mx,t_mx);
        }
        cout << mn << ' ' << mx << '\n';

    }
}
```

## Solution 2

### 시간 28 ms

max 함수 대신 if 로 조건 처리 후 값 갱신 

```c
#include "bits/stdc++.h"

using namespace std;

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int t;
    cin >> t;
    while (t--) {
        int stick , n, mx = -1, mn = -1;
        cin >> stick >> n;
        for (int i = 0; i < n; ++i) {
            int d;
            cin >> d;
           int t_mn = min(stick-d ,d);
           int t_mx = max(stick-d,d);
            if (t_mn > mn){
                mn = t_mn;
            }
            if (t_mx >mx){
                mx = t_mx;
            }
           mx = max(mx,t_mx);
        }
        cout << mn << ' ' << mx << '\n';

    }
}
```
