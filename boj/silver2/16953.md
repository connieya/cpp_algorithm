# BOJ 16953 A -> B

## solution 1

```c++
#include "bits/stdc++.h"

using namespace std;
typedef long long ll;
int ans = 40;

void dfs(ll a, ll b, int L) {
    if (a >= b) {
        if (a == b) {
            ans = min(L, ans);
        }
        return;
    }
    dfs(a * 10 + 1, b, L + 1);
    dfs(a * 2, b, L + 1);
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    ll a, b;
    cin >> a >> b;
    dfs(a, b, 1);
    cout << (ans == 40 ? -1 : ans);
}
```

## solution 2

역으로 생각하기 !!

B->A 로 갈 수 있는지

두 개의 연산을 보니깐 짝수 아니면 맨 뒷자리 수가 1인지 판단하면된다.

```c++
#include "bits/stdc++.h"

using namespace std;
typedef long long ll;

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    ll a, b;
    cin >> a >> b;
    for (int i = 1; a <= b; ++i) {
        if (a == b) {
            cout << i;
            return 0;
        }
        if (b % 2 == 0) {
            b /= 2;
        } else if (b % 10 == 1) {
            b = (b - 1) / 10;
        } else {
            break;
        }
    }
    cout << -1;
}
```
