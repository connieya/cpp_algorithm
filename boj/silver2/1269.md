# BOJ 1269 대칭 차집합

## solution 1

### 이진 검색

```c
#include "bits/stdc++.h"

using namespace std;

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n,m,ans = 0;
    cin >> n >> m;
    vector<int> a(n),b(m);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    for (int i = 0; i < m; ++i) {
        cin >> b[i];
    }
    sort(a.begin(),a.end());
    sort(b.begin(),b.end());
    for(int i : a){
        ans += !binary_search(b.begin(),b.end(),i);
    }
    for(int i : b){
        ans += !binary_search(a.begin(),a.end(),i);
    }
    cout << ans;

}
```

## solution 2

### set 사용

```c
#include "bits/stdc++.h"

using namespace std;

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n,m,a;
    cin >> n >> m;
    set<int> s;
    while (n--) {
        cin >> a;
        s.insert(a);
    }
    while (m--){
        cin  >> a;
        if(s.count(a)) {
            s.erase(a);
        }else {
            s.insert(a);
        };
    }
    cout << s.size();

}
```

## solution 3 

### 이분 탐색 vol.2

교집합 = k

합집합 =  n+m-k

대칭 차집합 = n+m-2*k

```c
#include "bits/stdc++.h"

using namespace std;

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n,m , k =0;
    cin >> n >> m;
    vector<int> a(n);
    for(auto &i : a){
        cin >> i;
    }
    sort(a.begin(),a.end());
    for (int i = 0; i < m; ++i) {
        int t;
        cin >> t;
        if(binary_search(a.begin(),a.end(),t)) k++;
    }
    cout << n+m-2*k;
}
```
