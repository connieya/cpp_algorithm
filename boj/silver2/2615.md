# BOJ 2615 오목

## solution 1

```c++
#include "bits/stdc++.h"

using namespace std;

int board[21][21];
int dx[] = {1, 0, 1, -1};
int dy[] = {0, 1, 1, 1};
pair<int, int> res;


int recur(int x, int y, int dx, int dy, int num) {
    if (x > 19 || x < 1 || y > 19 || y < 1) return 0;
    if (board[x][y] == num) {
        return 1 + recur(x + dx, y + dy, dx, dy, num);
    }
    return 0;
}


bool solve(int num, int x, int y) {
    for (int i = 0; i < 4; ++i) {
        bool flag = false;
        int nx = x + dx[i];
        int ny = y + dy[i];
        int sz = 0;
        while (nx <= 19 && nx >= 1 && ny <= 19 && ny >= 1 && board[nx][ny] == num) {
            nx += dx[i];
            ny += dy[i];
            sz++;
        }
        nx = x - dx[i];
        ny = y - dy[i];
        while (nx <= 19 && nx >= 1 && ny <= 19 && ny >= 1 && board[nx][ny] == num) {
            flag = true;
            sz++;
            nx -= dx[i];
            ny -= dy[i];
        }
        if (sz + 1 == 5) {
            if (flag) {
                res.first = nx+dx[i];
                res.second = ny+dy[i];
            } else {
                res.first = x;
                res.second = y;
            }
            return true;
        }
    }
    return false;
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    for (int i = 1; i <= 19; ++i) {
        for (int j = 1; j <= 19; ++j) {
            cin >> board[i][j];
        }
    }
    for (int i = 1; i <= 19; ++i) {
        for (int j = 1; j <= 19; ++j) {
            if (!board[i][j]) continue;
            if (solve(board[i][j], i, j)) {
                cout << board[i][j] << '\n';
                cout << res.first << " " << res.second;
                exit(0);
            }
        }
    }
    cout << 0;
}
```
