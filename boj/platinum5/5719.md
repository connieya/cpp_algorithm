# BOJ 5719 거의 최단 경로

## solution 1

```c
#include "bits/stdc++.h"
#define INF 20000000

using namespace std;

vector<pair<int,int>> graph[501];
vector<int> path[501];
int dist[501];
bool visited[501];


void dijkstra(int start){
    fill(dist,dist+501, INF);
    priority_queue<pair<int,int>> pq;
    dist[start] = 0;
    pq.push({0,start});
    while (!pq.empty()){
        auto cur = pq.top();
        pq.pop();
        int now = cur.second;
        int cost = -cur.first;
        if(dist[now] < cost) continue;
        for(auto nxt : graph[now]) {
            int next = nxt.first;
            int next_cost = nxt.second;
            if(dist[next] > next_cost+cost){
                dist[next] = next_cost+cost;
                path[next].clear();
                path[next].push_back(now);
                pq.push({-dist[next], next});
            }else if(dist[next] == next_cost+cost){
                path[next].push_back(now);
            }
        }

    }
}

void bfs(int start){
    queue<int> q;
    q.push(start);
    visited[start] = true;

    while (!q.empty()){
        int cur = q.front();
        q.pop();
        for(int prev : path[cur]){
            for(auto &nxt : graph[prev]){
                if(nxt.first == cur){
                    nxt.second = INF;

                    if(!visited[prev]){
                        q.push(prev);
                        visited[prev] = true;
                    }
                }
            }
        }
    }

}

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n,m,s,e,a,b,c;
    while (1){
        cin >> n >> m;
        if(!n && !m) break;
        cin >> s >> e;
        for (int i = 0; i < m; ++i) {
           cin >> a >> b >> c;
           graph[a].push_back({b,c});
        }
        dijkstra(s);
        bfs(e);
        dijkstra(s);
        cout << (dist[e] == INF ? -1 : dist[e] )<< '\n';
        for(int i=0; i<501; i++){
            graph[i].clear();
            path[i].clear();
            visited[i] = false;
            dist[i] = false;
        }
    }

}
```
