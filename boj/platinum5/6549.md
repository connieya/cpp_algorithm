# BOJ 6549 히스토그램에서 가장 큰 직사각형

## solution 1

```c++
#include "bits/stdc++.h"

using namespace std;
typedef long long ll;

ll arr[100001];

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int t;
    while (1){
        cin >> t;
        if (!t) break;
        ll ans = 0;
        stack<ll> stack;
        for (int i = 0; i < t; ++i) {
            cin >> arr[i];
            while (!stack.empty() && arr[stack.top()] > arr[i]){
                ll height = arr[stack.top()];
                stack.pop();
                ll width = i;
                if(!stack.empty()){
                    width = i - stack.top()-1;
                }
                if(ans < width*height){
                    ans = width*height;
                }

            }
            stack.push(i);
        }
        while (!stack.empty()){
            ll height = arr[stack.top()];
            stack.pop();
            ll width = t;
            if(!stack.empty()){
                width = t-stack.top()-1;
            }
            if(ans < width*height){
                ans = width*height;
            }
        }
        cout << ans << '\n';

    }
}
```

## solution 2

```c++
#include "bits/stdc++.h"

#define INF 100001
typedef long long ll;

using namespace std;

ll arr[INF];
int Left[INF];
int Right[INF];

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int sz;
    while (1) {
        ll ans = -1;
        cin >> sz;
        vector<pair<int, int>> stack;
        if (!sz) break;
        for (int i = 0; i < sz; ++i) {
            cin >> arr[i];
            while (!stack.empty() && stack.back().second >= arr[i]) {
                stack.pop_back();
            }
            Left[i] = stack.empty() ? -1 : stack.back().first;
            stack.push_back({i, arr[i]});
        }
        stack.clear();
        for (int i = sz - 1; i >= 0; --i) {
            while (!stack.empty() && stack.back().second >= arr[i]) {
                stack.pop_back();
            }
            Right[i] = stack.empty() ? sz : stack.back().first;
            stack.push_back({i, arr[i]});
        }
        stack.clear();
        for (int i = 0; i < sz; ++i) {
            ll t = arr[i] * (Right[i] - Left[i] - 1);
            ans = max(ans, t);
        }
        cout << ans << '\n';

    }
}
```

```c++
            s.push_back(i);
```
