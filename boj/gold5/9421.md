# BOJ 9421 소수상근수


## solution 1

```c
#include "bits/stdc++.h"

using namespace std;

bool isPrime[1000001];

void init(int n) {
    fill(isPrime, isPrime + n + 1, true);

    for (int i = 2; i * i <= n; i++) {
        if (!isPrime[i]) continue;
        for (int j = i * i; j <= n; j += i) {
            isPrime[j] = false;
        }
    }
}

bool solve(int num) {
    unordered_set<int> s;
    s.insert(num);
    while (1) {
        int t = 0;
        while (num) {
            int md = num % 10;
            t += md * md;
            num /= 10;
        }

        if (s.find(t) == s.end()) {
            s.insert(t);
        } else {
            return false;
        }
        if (t == 1) return true;
        num = t;
    }

}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n;
    cin >> n;
    init(n);
    for (int i = 2; i <= n; i++) {
        if (!isPrime[i]) continue;
        if (solve(i)) {
            cout << i << '\n';
        }
    }
}
```

## solution 2

### 시간 복잡도 92ms -> 8ms

```c
#include "bits/stdc++.h"

using namespace std;

bool isPrime[1000001];
int sg[1000001];

bool solve(int num) {
    if (num == 1) return sg[num] = 1;
    if (sg[num] != -1) return sg[num];
    sg[num] = 0;
    int next = 0, original = num;
    while (num) {
        next += (num % 10) * (num % 10);
        num /= 10;
    }
    return sg[original] = solve(next);
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n;
    cin >> n;
    memset(sg, -1, sizeof(sg));
    for (long long i = 2; i <= n; i++) {
        if (!isPrime[i]) {
            for (long long j = i * i; j <= n; j += i) {
                isPrime[j] = true;
            }
            if (solve(i)) {
                cout << i << '\n';
            }
        }

    }
}
```
