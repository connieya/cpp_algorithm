## Solution 1


### 정규 표현식

```c++
#include "bits/stdc++.h"

using namespace std;

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int t;
    cin >> t;
    regex r(R"((100+1+|01)+)");
    while (t--) {
        string str;
        cin >> str;
        regex_match(str, r) ? cout << "YES\n" : cout << "NO\n";
    }
}
```
