# BOJ 5557 1학년


## Solution 1

```c++
#include "bits/stdc++.h"
#define INF 101


typedef long long ll;

using namespace std;

ll arr[INF];
ll dp[INF][21];

ll func(int idx , int val) {
    if (idx < 0 || val < 0 || val > 20) return 0;
    if (idx == 0 && val == arr[0]) return 1;
    if (dp[idx][val] !=-1) return dp[idx][val];
    return dp[idx][val] = func(idx-1 , val-arr[idx]) + func(idx-1 , val+arr[idx]);
}

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n;
    cin >> n;
    for (int i = 0; i < n; ++i) {
        cin >> arr[i];
    }
    memset(dp,-1, sizeof(dp));
    cout << func(n-2,arr[n-1]);


}
```

## Solution 2

```c++
#include "bits/stdc++.h"
#define INF 101


typedef long long ll;

using namespace std;

int arr[INF];
ll dp[INF][21];


int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n;
    cin >> n;
    for (int i = 0; i < n; ++i) {
        cin >> arr[i];
    }
    dp[0][arr[0]] = 1;
    for (int i = 1; i < n - 1; ++i) {
        for (int j = 0; j <= 20-arr[i]; ++j) {
            dp[i][j] += dp[i-1][j+arr[i]];
        }
        for (int j = arr[i]; j <=20; ++j) {
            dp[i][j] += dp[i-1][j-arr[i]];
        }
    }
    cout << dp[n-2][arr[n-1]];
}
```

## Solution 3

```c++
#include "bits/stdc++.h"
#define INF 101


typedef long long ll;

using namespace std;

int arr[INF];
ll dp[INF][21];


int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n;
    cin >> n;
    for (int i = 0; i < n; ++i) {
        cin >> arr[i];
    }
    dp[0][arr[0]] = 1;
    for (int i = 1; i < n; ++i) {
        for (int j = 0; j <= 20; ++j) {
            for(const int nxt : {j+arr[i], j-arr[i]}){
                if(nxt <= 20 && nxt >= 0){
                    dp[i][nxt] += dp[i-1][j];
                }
            }
        }
    }
    cout << dp[n-2][arr[n-1]];
}
```
