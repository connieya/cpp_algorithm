# BOJ 17085 십자가 2개 놓기

## solution 1

```c++
#include "bits/stdc++.h"

using namespace std;

char board[16][16];
int n, m;
int dx[] = {-1, 0, 1, 0};
int dy[] = {0, 1, 0, -1};

bool OOP(int x, int y) {
    return x < 0 || x >= n || y < 0 || y >= m;
}

bool isCross(int x, int y, int sz) {
    for (int i = 0; i < 4; ++i) {
        int tmp = sz;
        int nx = x;
        int ny = y;
        while (tmp--) {
            nx += dx[i];
            ny += dy[i];
            if (OOP(nx, ny) || board[nx][ny] == '.') return 0;
        }
    }
    return 1;
}

bool isOverlap(tuple<int, int, int> &a, tuple<int, int, int> &b) {
    bool check[16][16] = {false,};
    int ax, ay, as;
    tie(ax, ay, as) = a;
    check[ax][ay] = true;
    for (int i = 0; i < 4; ++i) {
        int nx = ax;
        int ny = ay;
        for (int j = 0; j < as; ++j) {
            nx += dx[i];
            ny += dy[i];
            check[nx][ny] = true;
        }
    }

    int bx, by, bs;
    tie(bx, by, bs) = b;
    if (check[bx][by]) return false;
    for (int i = 0; i < 4; ++i) {
        int nx = bx;
        int ny = by;
        for (int j = 0; j < bs; ++j) {
            nx += dx[i];
            ny += dy[i];
            if (check[nx][ny]) return false;
        }
    }
    return true;
}


int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    cin >> n >> m;
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            cin >> board[i][j];
        }
    }
    vector<tuple<int, int, int>> cross;
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            if (board[i][j] == '.') continue;
            for (int k = 0; isCross(i, j, k); k++) {
                cross.push_back({i, j, k});
            }
        }
    }
    int len = cross.size();
    int ans = 0;
    for (int i = 0; i < len; ++i) {
        for (int j = i + 1; j < len; ++j) {
            if (isOverlap(cross[i], cross[j])) {
                int ax, ay, as, bx, by, bs;
                tie(ax, ay, as) = cross[i];
                tie(bx, by, bs) = cross[j];
                ans = max(ans, (as * 4 + 1) * (bs * 4 + 1));
            }
        }
    }
    cout << ans;
}
```
