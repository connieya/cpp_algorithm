# BOJ 15686 치킨 배달

## solution 1

### 시간 4ms

```c
#include "bits/stdc++.h"

using namespace std;
int n,r;
bool visited[50];
vector<pair<int ,int>> comp;
vector<pair<int ,int>> city;
int res = 2145000;

void dfs(int L , int start){
    if (L == r) {
        int sum = 0;
        for (int i = 0; i < city.size(); ++i) {
            auto city_dis = city[i];
            int MIN = 2147000;
            for (int j = 0; j < comp.size(); ++j) {
                if (visited[j]) {
                    auto chicken = comp[j];
                    int dis = abs(city_dis.first-chicken.first) + abs(city_dis.second-chicken.second);
                    MIN = min(MIN , dis);
                }
            }
            sum += MIN;
        }
        res = min(res ,sum);
        return;
    }
    for (int i = start; i < comp.size(); ++i) {
        visited[i] = true;
        dfs(L+1,i+1);
        visited[i] = false;
    }
}

int main(void)
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int tmp;
    cin >> n >> r;
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            cin >> tmp;
            if (tmp == 1){
                city.push_back({i,j}); continue;
            }
            if (tmp == 2){
                comp.push_back({i,j});
            }
        }
    }
    dfs(0,0);
    cout << res;
}
```

## solution 2

### 시간 8ms

```c
#include "bits/stdc++.h"

using namespace std;

int n, m;
int arr[51][51];
int ans = INT_MAX;
bool check[51][51];
vector<pair<int, int>> chicken;
vector<pair<int, int>> choice;

void dfs(int l, int start) {
    if (l == m) {
        int sum = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                if (arr[i][j] == 1) {
                    int mn = INT_MAX;
                    for (int k = 0; k < choice.size(); ++k) {
                        mn = min(mn, abs(i - choice[k].first) + abs(j - choice[k].second));
                    }
                    sum += mn;
                }
            }
        }
        if (ans > sum) ans = sum;

        return;
    }

    for (int i = start; i < chicken.size(); ++i) {
        choice.push_back({chicken[i].first, chicken[i].second});
        dfs(l + 1, i + 1);
        choice.pop_back();
    }
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    cin >> n >>  m;
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            cin >> arr[i][j];
            if (arr[i][j] == 2) {
                chicken.push_back({i, j});
            }
        }
    }
    dfs(0, 0);
    cout << ans;
}
```

## solution 3

### 시간 32ms

```c
#include "bits/stdc++.h"

using namespace std;


int arr[51][51];
vector<pair<int, int>> chicken;
vector<pair<int,int>> person;

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n,m;
    cin >> n >>  m;
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            cin >> arr[i][j];
            if (arr[i][j] == 2) {
                chicken.push_back({i, j});
            }else if(arr[i][j] == 1){
                person.push_back({i,j});
            }
        }
    }
    vector<int> d(chicken.size());
    for (int i = 0; i < m; ++i) {
        d[i] = 1;
    }
    sort(d.begin(),d.end());
    int ans = INT_MAX;
    do {
        int sum = 0;
        for(auto p : person){
            vector<int> dist;
            for (int i = 0; i < chicken.size(); ++i) {
                if(!d[i]) continue;
                dist.push_back(abs(chicken[i].first-p.first)+ abs(chicken[i].second-p.second));
            }
            sum += *min_element(dist.begin(),dist.end());
        }
        if(ans > sum) ans = sum;
    } while (next_permutation(d.begin(),d.end()));

    cout << ans;
}
```
