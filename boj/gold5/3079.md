# BOJ 3079 입국심사

결정 알고리즘!!

k초 이하의 시간으로 입국 심사를 할 수 있는가??

최적해를 구하기 위해 이분탐색을 하면 됨

## Solution1 

```c++
#include "bits/stdc++.h"

typedef long long ll;

using namespace std;

ll arr[100001];

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    ll n, m, lt, rt = 0 ,ans =0;
    cin >> n >> m;
    for (int i = 0; i < n; ++i) {
        cin >> arr[i];
        if (arr[i] > rt) {
            rt = arr[i];
        }
    }
    rt *= m;
    lt = 0;
    while (lt <= rt) {
        ll mid = (lt+rt)/2;
        ll total = 0;
        for (int i = 0; i < n; ++i) {
            total += mid/arr[i];
        }
        if (total >= m){
            rt = mid-1;
            ans = mid;
        }else {
            lt = mid+1;
        }
    }
    cout << ans;
}
```

## Solution 2

```c++
#include "bits/stdc++.h"

using namespace std;

typedef long long ll;
const ll INF = ll(1e18);
int n, m;
vector<int> v;

bool check(ll k) {
    ll cnt = 0;
    for (int i = 0; i < n; ++i) {
        cnt += k / v[i];
        if (cnt >= m) return true;
    }
    return false;
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    cin >> n >> m;
    v.resize(n);
    for (int i = 0; i < n; ++i) {
        cin >> v[i];

    }
    ll lt = 0, rt = INF;
    while (lt + 1 < rt) {
        ll mid = (lt + rt) / 2;
        if (check(mid)){
            rt = mid;
        }else {
            lt = mid;
        }
    }
    cout << rt;
}
```
