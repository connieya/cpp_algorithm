# BOJ 14469 소가 길을 건너간 이유 3

## solution 1

```c++
#include "bits/stdc++.h"

using namespace std;

bool comp(pair<int, int> a, pair<int, int> b) {
    return a.first < b.first;
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n, a, b;
    cin >> n;
    vector<pair<int, int>> v;
    for (int i = 0; i < n; ++i) {
        cin >> a >> b;
        v.push_back({a, b});
    }
    sort(v.begin(), v.end(), comp);
    int time = v[0].first+v[0].second;
    for (int i = 1; i < n; ++i) {
        int a = v[i].first;
        int b = v[i].second;
        if(time <= a){
            time = a;
        }
        time += b;
    }
    cout << time;
}
```
