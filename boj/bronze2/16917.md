# BOJ 16917 양념 반 후라이드 반

## solution 1

```c
#include "bits/stdc++.h"

using namespace std;

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int a,b,c,x,y;
    cin >> a >> b >> c >> x >> y;
    int mx = x;
    int ans = INT_MAX;
    mx = max(mx,y);
    for (int i = 0; i <= mx; ++i) {
        int nx = x-i < 0 ? 0 :x-i;
        int ny = y-i < 0 ? 0 : y-i;
        ans = min(ans, a*nx+b*ny+c*(i*2));
    }
    cout << ans;
}
```

## solution 2

### 브루트 포스 , 음수 처리 센스

```c
#include "bits/stdc++.h"

using namespace std;

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int a,b,c,x,y;
    cin >> a >> b >> c >> x >> y;
    int ans = x*a+b*y;
    for (int i = 1; i <= 100000; ++i) {
        int price = 2*i*c + max(0,x-i)*a + max(0,y-i)*b;
        if(ans > price) ans = price;
    }
    cout << ans;
}
```
