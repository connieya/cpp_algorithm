# BOJ 11066 파일 합치기

## solution 1

### 메모리 : 3008KB , 시간 : 300ms

```c
#include "bits/stdc++.h"

using namespace std;

int n;
int arr[502];
int dp[502][502];

int solve(int i, int j) {
    if(i==j) return 0;

    if(dp[i][j] != -1) return dp[i][j];
    int &ans = dp[i][j];
    int sum = 0;
    for (int k = i; k <=j ; ++k) {
        sum += arr[k];
    }

    for(int k=i; k<=j-1; k++){
        int tmp = solve(i,k) + solve(k+1,j)+sum;
        if(ans == -1 || ans > tmp){
            ans = tmp;
        }
    }
    return ans;
}

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int T;
    cin >> T;
    while (T--) {
       cin >> n;
        memset(dp,-1, sizeof(dp));
        for (int i = 1; i <= n; ++i) {
            cin >> arr[i];
        }
        cout << solve(1,n) << '\n';
    }
}
```

## solution 2

### 메모리 : 3008KB , 시간 : 136ms

```c
#include "bits/stdc++.h"
#define INF 50000001
using namespace std;

int n;
int arr[502];
int dp[502][502];


int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int T;
    cin >> T;
    while (T--) {
        cin >> n;
        vector<int> sum(n + 1);
        for (int i = 1; i <= n; ++i) {
            cin >> arr[i];
            sum[i] = sum[i - 1] + arr[i];
        }
        for (int i = 1; i <= n; ++i) {
            for (int j = 1; j <=n-i; ++j) {
                dp[j][i+j] = INF;
                for (int k = j; k < i+j; ++k) {
                    dp[j][i+j] = min(dp[j][i+j] , dp[j][k]+dp[k+1][i+j]+sum[i+j]-sum[j-1]);
                }
            }
        }
        cout << dp[1][n] << '\n';
        memset(dp, 0, sizeof(dp));
    }
}
```

## solution 4

```c
#include "bits/stdc++.h"

using namespace std;

int n;
int arr[502];
int sum[502];
int dp[502][502];

int solve(int i, int j) {
    if(i==j) return 0;
    if(i+1 == j) return arr[i]+arr[j];
    int &ans = dp[i][j];
    if(ans != -1) return ans;
    ans = int(1e9);
    for(int k=i; k<j; k++){
       ans = min(ans, solve(i,k)+ solve(k+1,j));
    }
    return ans+=sum[j]-sum[i-1];
}

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int T;
    cin >> T;
    while (T--) {
        cin >> n;
        memset(dp,-1, sizeof(dp));
        for (int i = 1; i <= n; ++i) {
            cin >> arr[i];
            sum[i] = sum[i-1]+arr[i];
        }
        cout << solve(1,n) << '\n';
        memset(sum,0, sizeof(sum));
    }
}
```
