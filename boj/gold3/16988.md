# BOJ 16988 Baaaaaaaaaduk2(Easy)

## solution 1

### 시간 652 ms

```c++
#include "bits/stdc++.h"

using namespace std;

int n, m;
int board[21][21];
bool check[21][21];

int dx[] = {-1, 0, 1, 0};
int dy[] = {0, 1, 0, -1};
int ans;

int bfs(int x, int y) {
    queue<pair<int, int>> q;
    check[x][y] = true;
    q.push({x, y});
    int cnt = 1;
    bool flag = false;
    while (!q.empty()) {
        x = q.front().first;
        y = q.front().second;
        q.pop();
        for (int i = 0; i < 4; ++i) {
            int nx = x + dx[i];
            int ny = y + dy[i];
            if (nx < 0 || nx >= n || ny < 0 || ny >= m) continue;
            if (board[nx][ny] == 0) flag = true;
            if (board[nx][ny] == 2 && !check[nx][ny]) {
                check[nx][ny] = true;
                q.push({nx, ny});
                cnt++;
            }

        }
    }
    if(flag) return 0;
    return cnt;
}

int baduk() {
    memset(check, false, sizeof(check));
    int ans = 0;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; ++j) {
            if (board[i][j] == 2 && !check[i][j]) {
                ans += bfs(i, j);
            }
        }
    }
    return ans;
}

void dfs(int x, int y, int L) {
    if (L == 2) {
        ans = max(ans, baduk());
        return;
    }

    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            if (board[i][j]) continue;
            board[i][j] = 1;
            dfs(i, j, L + 1);
            board[i][j] = 0;
        }
    }
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    cin >> n >> m;
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            cin >> board[i][j];
        }
    }
    dfs(0, 0, 0);
    cout << ans;
}
```
