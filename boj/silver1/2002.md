# BOJ 2002 추월


## Solution 1

```c
#include "bits/stdc++.h"

using namespace std;

string arr[1001];

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    unordered_map<string, int> mp;
    int n, ans = 0;
    string car_num;
    cin >> n;
    for (int i = 0; i < n; ++i) {
        cin >> car_num;
        mp[car_num] = i;
    }
    for (int i = 0; i < n; ++i) {
        cin >> arr[i];
    }
    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            if (mp[arr[i]] > mp[arr[j]]){
                ans++;
                break;
            }
        }
    }
    cout << ans;

}
```
