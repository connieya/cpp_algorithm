# BOJ 3896 소수 사이 수열


## solution 1

## 에라토스테네스의 체 , 구현

```c
#include "bits/stdc++.h"

#define INF 2000000
using namespace std;

bool isPrime[INF];

void init() {
    memset(isPrime, true, sizeof(isPrime));
    for (int i = 2; i * i < INF; i++) {
        if (!isPrime[i]) continue;
        for (int j = i * i; j < INF; j += i) {
            isPrime[j] = false;
        }
    }
}

int solve(int start) {
    int tmp = start;
    while (1) {
        tmp--;
        if (isPrime[tmp]) break;
    }
    while (1) {
        start++;
        if (isPrime[start]) break;
    }
    return start - tmp;
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int t, v;
    cin >> t;
    init();
    while (t--) {
        cin >> v;
        if (isPrime[v]) {
            cout << 0 << '\n';
            continue;
        }
        cout << solve(v) << '\n';
    }
}
```

## solution 2

에라토스 테네스의 체 사용할 필요 없음

```c
#include "bits/stdc++.h"

using namespace std;

bool isPrime(int num) {
    for (int i = 2; i * i <= num; i++) {
        if (num % i == 0) return false;
    }
    return true;
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int t, v;
    cin >> t;
    while (t--) {
        cin >> v;
        int x = v, y = v;
        while (!isPrime(x)) x--;
        while (!isPrime(y)) y++;
        cout << y - x << '\n';
    }
}
```
