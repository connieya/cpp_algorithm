# BOJ 10597 순열장난

## Solution 1

```c++
#include "bits/stdc++.h"

using namespace std;

string str;
int max_num;
vector<int> answer;
bool checked[51];

void dfs(int L) {
    if (L >= str.sz()) {
        for (auto ans: answer) {
            cout << ans << ' ';
        }
        exit(0);
    }
    int num1 = str[L] - '0';
    if (!checked[num1] && num1) {
        checked[num1] = true;
        answer.push_back(num1);
        dfs(L + 1);
        answer.pop_back();
        checked[num1] = false;
    }

    if (L == str.sz() - 1) return;
    int num2 = (str[L] - '0') * 10 + str[L + 1] - '0';
    if (num2 <= max_num && !checked[num2]) {
        checked[num2] = true;
        answer.push_back(num2);
        dfs(L + 2);
        answer.pop_back();
        checked[num2] = false;
    }

}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    cin >> str;
    int len = str.length();
    if (len <= 9) {
        for (char ch: str) {
            cout << ch << ' ';
        }
        exit(0);
    }
    max_num = (len - 9) / 2 + 9;
    dfs(0);
}
```
