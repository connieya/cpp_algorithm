# BOJ 11660 구간 합 구하기 5


## Solution 1

### 구간 합 / 메모리 : 6124 KB , 시간 : 184 ms

```c++
#include "bits/stdc++.h"

#define MX 1025

using namespace std;

int arr[MX][MX];

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n, m;
    cin >> n >> m;
    for (int i = 1; i <= n; ++i) {
        for (int j = 1; j <= n; ++j) {
            cin >> arr[i][j];
        }
    }
    for (int i = 2; i <= n; ++i) {
        for (int j = 1; j <= n; ++j) {
            arr[i][j] += arr[i - 1][j];
        }
    }
    for (int i = 0; i < m; ++i) {
        int x1, y1, x2, y2;
        cin >> x1 >> y1 >> x2 >> y2;
        int ans = 0;
        for (int j = y1; j <= y2; ++j) {
            ans += arr[x2][j] - arr[x1-1][j];
        }
        cout << ans << '\n';
    }
}
```

## Solution 2


###  메모리 : 6124  KB  /  시간 : 128 ms

```c++
#include "bits/stdc++.h"

#define MX 1025

using namespace std;

int arr[MX][MX];

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n, m;
    cin >> n >> m;
    for (int i = 1; i <= n; ++i) {
        for (int j = 1; j <= n; ++j) {
            cin >> arr[i][j];
            arr[i][j] += arr[i - 1][j] + arr[i][j - 1] - arr[i - 1][j - 1];
        }
    }
    for (int i = 0; i < m; ++i) {
        int x1, y1, x2, y2;
        cin >> x1 >> y1 >> x2 >> y2;
        cout << arr[x2][y2] - arr[x1 - 1][y2] - arr[x2][y1 - 1] + arr[x1 - 1][y1 - 1] << '\n';
    }
}
```
