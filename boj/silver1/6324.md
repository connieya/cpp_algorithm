# BOJ 6324 URLs

## Solution 1

```c
#include "bits/stdc++.h"

using namespace std;

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    regex r(R"((^\w+)://([^:/]+)(:\d+)?(/.+$)?)");
    int n;
    cin >> n;
    for (int i = 1; i <= n; ++i) {
        string url, ans[4];
        cin >> url;
        auto it = *sregex_iterator(url.begin(), url.end(), r);
        for (int i = 0; i < 4; i++) {
            ans[i] = it[i + 1];
            if (ans[i].empty()) ans[i] = "<default>";
        }
        if (ans[2][0] == ':') ans[2] = ans[2].substr(1);
        if (ans[3][0] == '/') ans[3] = ans[3].substr(1);

        cout << "URL #" << i << '\n';
        cout << "Protocol = " << ans[0] << '\n';
        cout << "Host     = " << ans[1] << '\n';
        cout << "Port     = " << ans[2] << '\n';
        cout << "Path     = " << ans[3] << '\n' << '\n';
    }
}
```
