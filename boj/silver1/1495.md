# BOJ 1495 기타리스트

## solution 1


```c
#include "bits/stdc++.h"

using namespace std;

int n,s,m;
int arr[52];
bool check[52][1001];

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    cin >> n >> s >> m;
    for (int i = 1; i <= n; ++i) {
        cin >> arr[i];
    }
        check[0][s] = true;
    for (int i = 1; i <= n; ++i) {
        for (int j = 0; j <= m; ++j) {
            if (check[i-1][j]) {
                if (j+arr[i] <= m){
                    check[i][j+arr[i]] = true;
                }
                if (j-arr[i] >= 0){
                    check[i][j-arr[i]] = true;
                }
            }
        }
    }
    int ans = -1;
    for (int i = 0; i <=m; ++i) {
        if(check[n][i]){
            ans = max(ans,i);
        }
    }
    cout << ans;
}
```
