# BOJ 16174 점프왕 쩰리(Large)

## Solution 1

### DFS

```c++
#include "bits/stdc++.h"

using namespace std;

int n;
int board[65][65];
bool visited[65][65];

bool dfs(int x, int y) {
    if (x == n-1 && y == n-1){
        return true;
    }
    int jump = board[x][y];
    if (jump + x < n && !visited[jump + x][y]) {
        visited[jump+x][y] = true;
        if (dfs(jump + x, y)) return true;
    }
    if(jump+y < n && !visited[x][jump+y]){
        visited[x][jump+y] = true;
        if (dfs(x,jump+y)) return true;
    }
    return false;
}

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    cin >> n;
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            cin >> board[i][j];
        }
    }
    dfs(0,0) ? cout << "HaruHaru" : cout << "Hing";
}
```

## Solution 2

### DP

```c++
#include "bits/stdc++.h"

using namespace std;

int n, x;
int dp[65][65] = {1};

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    cin >> n;
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            cin >> x;
            if (dp[i][j] == 0) continue;
            if (i + x < n) {
                dp[i + x][j] = 1;
            }
            if (j + x < n) {
                dp[i][j + x] = 1;
            }
        }
    }
    dp[n-1][n-1] ? cout << "HaruHaru" : cout << "Hing";

}
```
