# BOJ 16967 배열 복원하기

## solution 1

```c
#include "bits/stdc++.h"

using namespace std;

int x, y, w, h;
int arr[601][601];
int ans[301][301];

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    cin >> x >> y >> w >> h;
    for (int i = 0; i < x + w; ++i) {
        for (int j = 0; j < y + h; ++j) {
            cin >> arr[i][j];
        }
    }

    for (int i = 0; i < x; ++i) {
        for (int j = 0; j < y; ++j) {
            if (i < w || j < h) {
                ans[i][j] = arr[i][j];
            } else {
                ans[i][j] = arr[i][j] - ans[i - w][j - h];
            }
        }
    }

    for (int i = 0; i < x; ++i) {
        for (int j = 0; j < y; ++j) {
            cout << ans[i][j] << ' ';
        }
        cout << '\n';
    }

}
```
