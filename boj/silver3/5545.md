# BOJ 5545 최고의 피자

## solution 1

```c
#include "bits/stdc++.h"

using namespace std;

int arr[101];

bool cmp(int a, int b) {
    return a > b;
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n, a, b, c;
    cin >> n >> a >> b >> c;
    for (int i = 0; i < n; ++i) {
        cin >> arr[i];
    }
    sort(arr, arr + n, cmp);
    int sum = 0, ans = 0;
    for (int i = 0; i < n; ++i) {
        sum += arr[i];
        int tmp = (sum + c) / (a + (i + 1) * b);
        ans = max(tmp, ans);
    }
    ans = max(ans,c/a);
    cout << ans;
}
```
