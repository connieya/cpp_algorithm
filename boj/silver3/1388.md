# BOJ 1388 바닥 장식

## solution 1

```c
#include "bits/stdc++.h"

using namespace std;


char board[51][51];
bool visited[51][51];
int n, m;

void row(int x, int y, char z) {
    visited[x][y] = true;
    y++;
    while (y < m) {
        if (board[x][y] != z) break;
        visited[x][y] = true;
        y++;
    }
}

void column(int x, int y, char z) {
    visited[x][y] = true;
    x++;
    while (x < n) {
        if (board[x][y] != z) break;
        visited[x][y] = true;
        x++;
    }
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    cin >> n >> m;
    int ans = 0;
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            cin >> board[i][j];
        }
    }
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            if (visited[i][j]) continue;
            ans++;
            if (board[i][j] == '-') {
                row(i, j, board[i][j]);
            } else {
                column(i, j, board[i][j]);
            }
        }
    }
    cout << ans;
}
```
