# BOJ 1213 팰린드롬 만들기

## solution 1

```c++
#include "bits/stdc++.h"

using namespace std;



int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    string str;
    cin >> str;
    int alpha[26] = {0};

    for(auto ch : str){
        alpha[ch-'A']++;
    }
    int odd = 0;
    int idx = -1;
    for(int i=0; i<26; i++){
        if(alpha[i] && alpha[i]%2){
            odd++;
            idx = i;
        }
    }
    if(odd > 1){
        cout << "I'm Sorry Hansoo";
        return 0;
    }

    for (int i = 0; i < 26; ++i) {
        if(!alpha[i]) continue;
        for (int j = 0; j < alpha[i]/2; ++j) {
            cout << (char)(i+'A');
        }
    }
    if(idx != -1){
        cout << (char)(idx+'A');
    }
    for (int i = 25; i >=0; i--) {
        if(!alpha[i]) continue;
        for (int j = 0; j < alpha[i]/2; ++j) {
            cout << (char)(i+'A');
        }
    }
}
```
