# BOJ 2659 십자카드 문제

### 브루트 포스 , 구현

## Solution 1

```c++
#include "bits/stdc++.h"

using namespace std;

int a, b, c, d;

int calc() {
    vector<int> v;
    v.push_back(a * 1000 + b * 100 + c * 10 + d);
    v.push_back(b * 1000 + c * 100 + d * 10 + a);
    v.push_back(c * 1000 + d * 100 + a * 10 + b);
    v.push_back(d * 1000 + a * 100 + b * 10 + c);
    sort(v.begin(), v.end());
    return v[0];
}

bool isClockNum(int num) {
    int a = num / 1000;
    int b = num % 1000 / 100;
    int c = num % 100 / 10;
    int d = num % 10;
    if (d == 0 || c == 0 || b == 0) return false;
    if (num > b * 1000 + c * 100 + d * 10 + a) return false;
    if (num > c * 1000 + d * 100 + a * 10 + b) return false;
    if (num > d * 1000 + a * 100 + b * 10 + c)return false;
    return true;
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    cin >> a >> b >> c >> d;
    int target = calc();
    int ans = 0;
    for (int i = 1111; i < target; i++) {
        ans += isClockNum(i);
    }
    cout << ans + 1;
}
```

## Solution 2

### 시계수를 푸는 정석 , 센스 있는 코드

```c++
#include "bits/stdc++.h"

using namespace std;


int clock_num(int value) {
    int min = value;
    for (int i = 0; i < 3; ++i) {
        value = value % 1000 * 10 + value / 1000;
        if (min > value) {
            min = value;
        }
    }
    return min;
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int a,b,c,d;
    cin >> a >> b >> c >> d;
    int target = clock_num(a*1000+b*100+c*10+d);
    int ans = 0;
    for(int i = 1111; i <= target; i++){
        ans += clock_num(i) == i;
    }
    cout << ans;

}
```
