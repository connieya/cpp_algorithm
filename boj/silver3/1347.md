# BOJ 1347 미로 만들기

## solution 1

### 깰껌~

```c
#include "bits/stdc++.h"

using namespace std;

int n;
string str;
int board[101][101];
int dir = 3;

int dx[] = {0, -1, 0, 1}; // 동 , 북 , 서 , 남
int dy[] = {1, 0, -1, 0};

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    cin >> n;
    cin >> str;
    int x = 50, y = 50;
    board[x][y] = 1;
    int x1,y1,x2,y2;
    x1 = y1 = x2 = y2 = 50;
    for (auto ch: str) {
        if (ch == 'R') {
            --dir &= 3;
        } else if (ch == 'L') {
            ++dir &= 3;
        } else {
            x += dx[dir];
            y += dy[dir];
            board[x][y] = 1;
            x1 = min(x1, x);
            x2 = max(x2, x);
            y1 = min(y1, y);
            y2 = max(y2, y);
        }
    }

    for (int i = x1; i <= x2; i++) {
        for (int j = y1; j <= y2; j++) {
            cout << (board[i][j] ? '.' : '#');
        }
        cout << '\n';
    }
}
```

## solution 2

```c
#include "bits/stdc++.h"

using namespace std;

int n;
string str;
int board[101][101];
int dir = 3;

int dx[] = {0, -1, 0, 1}; // 동 , 북 , 서 , 남
int dy[] = {1, 0, -1, 0};

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    cin >> n;
    cin >> str;
    int x = 50, y = 50;
    board[x][y] = 1;
    int x1, y1, x2, y2;
    x1 = y1 = x2 = y2 = 50;
    for (auto ch: str) {
        if (ch == 'R') {
            dir--;
            dir = dir < 0 ? 3 : dir;
        } else if (ch == 'L') {
            dir = (dir + 1) % 4;
        } else {
            x += dx[dir];
            y += dy[dir];
            board[x][y] = 1;
            x1 = min(x1, x);
            x2 = max(x2, x);
            y1 = min(y1, y);
            y2 = max(y2, y);
        }
    }

    for (int i = x1; i <= x2; i++) {
        for (int j = y1; j <= y2; j++) {
            cout << (board[i][j] ? '.' : '#');
        }
        cout << '\n';
    }
}
```
