# BOJ 20291 파일 정리

## solution 1

```c++
#include "bits/stdc++.h"

using namespace std;

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n;
    cin >> n;
    string file;
    unordered_map<string,int> mp;
    for (int i = 0; i < n; ++i) {
        cin >> file;
        bool flag = false;
        string f;
        for(auto c : file){
            if(flag) f +=c;
            if(c == '.') flag = true;
        };
        mp[f]++;
    }
    vector<string> v;
    for(auto m : mp){
        v.push_back(m.first);
    }
    sort(v.begin(),v.end());
    for(auto a : v){
        cout << a << ' ' << mp[a] << '\n';
    }
}
```

## solution 2

```c++
#include "bits/stdc++.h"

using namespace std;

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n;
    cin >> n;
    string file;
    vector<string> v;
    for (int i = 0; i < n; ++i) {
        cin >> file;
        v.push_back(file.substr(file.find('.')+1));
    }
    sort(v.begin(),v.end());
    for (int i = 0; i < n;) {
        int j= i+1;
        int cnt = 1;
        while (j <n && v[i] == v[j]){
            cnt++;
            j++;
        }
        cout << v[i] << ' ' << cnt << '\n';
        i = j;
    }
}
```
