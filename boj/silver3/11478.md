# BOJ 11478 서로 다른 부분 문자열의 개수

## Solution 1

```c
#include "bits/stdc++.h"

using namespace std;

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    unordered_set<string> s;
    string str;
    cin >> str;
    int len = str.length();
    for (int i = 0; i < len; ++i) {
        string tmp = "";
        for (int j = i; j < len; ++j) {
            tmp += str[j];
            s.insert(tmp);
        }
    }
    cout << s.size();
}
```

    cout.tie(0);
    unordered_set<string> s;
    string str;
    cin >> str;
    int len = str.length();
    for (int i = 0; i < len; ++i) {
        for (int j = 1; j <= len - i; ++j) {
            s.insert(str .substr(i, j));
        }
    }
    cout << s.size();
}

## Solution 3

```c
#include "bits/stdc++.h"

using namespace std;

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    vector<string> s;
    string str;
    cin >> str;
    int len = str.length();
    for (int i = 0; i < len; ++i) {
        for (int j = 1; j <= len - i; ++j) {
            s.push_back(str .substr(i, j));
        }
    }
    sort(s.begin(),s.end());
    s.erase(unique(s.begin(),s.end()),s.end());
    cout << s.size();
}
```
