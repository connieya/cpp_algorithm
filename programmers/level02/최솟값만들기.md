# 프로그래머스 Lv.2 최솟값 만들기


## solution 1

```c
#include <bits/stdc++.h>

using namespace std;

int solution(vector<int> A, vector<int> B)
{
    int answer = 0;
    int len = B.size();
    sort(A.begin(), A.end());
    sort(B.begin(), B.end());
    for(int i=0; i< len; i++) {
        answer += A[i]*B[len-1-i];
    }
    return answer;
}
```
