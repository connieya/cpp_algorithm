# 프로그래머스 Lv.2 롤케이크 자르기

## solution 1

```c++
#include <bits/stdc++.h>

using namespace std;

int solution(vector<int> topping) {
    int answer = 0;
    unordered_map<int,int> mp;
    unordered_map<int,int> mp2;
    for(auto t : topping){
        mp[t]++;
    }
    int prev = 0;
    while(prev < topping.size()){
        int p = topping[prev];
        mp2[p]++;
        mp[p]--;
        if(mp[p] == 0) mp.erase(p);
        prev++;
        if(mp.size() == mp2.size()) answer++;
    }
    return answer;
}
```
