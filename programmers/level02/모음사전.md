# 프로그래머스 Lv.2 모음사전

## solution 1

```c
#include <bits/stdc++.h>

using namespace std;

const int length_of_word = 5;

int solution(string word) {
    int answer = 0;
    int num = 5;
    int mx = 0;
    int five_pow[5];
    for(int i=0; i< length_of_word; i++){
        mx += num;
        five_pow[i] = num;
        num *= 5;
       
    }
    for(int i=0; i< word.length(); i++) {
        char ch = word[i];
        if(ch == 'A'){
            answer += 1;
        }else if(ch == 'E') {
            answer += (mx / five_pow[i]) + 1;
        }else if(ch == 'I') {
            answer += (mx / five_pow[i])*2 + 1;
        }else if(ch == 'O') {
            answer += (mx / five_pow[i])*3 + 1;
        }else {
            answer += (mx / five_pow[i])*4 + 1;
        }
    }
    return answer;
}
```

## solution 2

```c
#include <bits/stdc++.h>

using namespace std;

string str = "AEIOU";
vector<string> v;

void dfs(int L , string ss) {
    if(L == 5){
        return;
    }
    
    for(int i=0; i< str.size(); i++) {
        v.push_back(ss+str[i]);
        dfs(L+1, ss+str[i]);
    }
}

int solution(string word) {
    dfs(0, "");
    int len = v.size();
    for(int i=0; i < len; i++){
        if(v[i] == word) return i+1;
    }
}
```
