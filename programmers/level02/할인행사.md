# 프로그래머스 Lv.2 할인 행사

## solution 1

```c++
#include <bits/stdc++.h>

using namespace std;

unordered_map<string,int> mp;
unordered_map<string,int> mp2;

bool isValid(vector<string> &want) {
    for(auto w : want){
        if(mp[w] > mp2[w]) return false;
    }
    return true;
}

int solution(vector<string> want, vector<int> number, vector<string> discount) {
    int answer = 0;
  
    for(int i=0; i<want.size(); i++){
        mp[want[i]] = number[i];
    }
    for(int i=0; i<10; i++){
        mp2[discount[i]]++;
    }
    answer += isValid(want);
    int start = 0 , end = 10;
    while(end < discount.size()) {
        mp2[discount[start]]--;
        mp2[discount[end]]++;
        answer += isValid(want);
        start++;
        end++;
    }
    return answer;
}
```
