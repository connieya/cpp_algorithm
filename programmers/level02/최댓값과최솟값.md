# 프로그래머스 Lv.2 최댓값과 최솟값


## solution 1

```c
#include <bits/stdc++.h>

using namespace std;

string solution(string s) {
    string answer = "";
    stringstream iss(s);
    vector<int> v;
    while(iss >> s){
        v.push_back(stoi(s));
    }
    sort(v.begin(),v.end());
    for(auto i : v){
        cout << i << ' ';
    }
    answer+= to_string(v[0]);
    answer+= " ";
    answer+= to_string(v[v.size()-1]);
    return answer;
}
```
