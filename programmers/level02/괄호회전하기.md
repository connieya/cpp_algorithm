# 프로그래머스 Lv.2 괄호 회전하기

## solution 1

```c
#include <bits/stdc++.h>

using namespace std;

bool isValid(char prev , char ch) {
    if(ch == ']' && prev != '['){
        return false;
    }else if(ch == ')' && prev != '(') {
        return false;
    }else if(ch == '}' && prev != '{' ){
        return false;
    }
    return true;
}


bool solve(string str) {
    stack<char> s;
    for(auto ch : str) {
        if(ch == '[' || ch == '(' || ch == '{'){
            s.push(ch);
        }else {
            if(s.size()) {
                bool res = isValid(s.top(), ch);
                if(res) s.pop();
                else return false;
            }else {
                return false;
            }
        }
    }
    
    if(s.size()) return false;
    return true;
    
}

int solution(string s) {
    int answer = 0;
    int len = s.size();
    for(int i=0; i<len; i++) {
        s.push_back(s.front());
        s.erase(s.begin());
        answer += solve(s);
    }
    return answer;
}
```
