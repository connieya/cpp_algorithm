# 프로그래머스 Lv.2 가장 큰 수

## solution 1

```c
#include <bits/stdc++.h>

using namespace std;

string solution(vector<int> numbers) {
    vector<string> v(numbers.size());
    for(int i=0; i< numbers.size(); i++){
        v[i] = to_string(numbers[i]);
    }
    
    sort(v.begin() , v.end() ,[&](string a ,string b){
       return a+b > b+a; 
    });
    string ret;
    for(auto r : v) ret+=r;
    
    return (ret[0] == '0') ? "0" : ret;
}
```
