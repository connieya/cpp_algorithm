# 프로그래머스 Lv.2 구명 보트

## solution 1

```c++
#include <bits/stdc++.h>

using namespace std;

// 최대 2명 밖에 태우지 못한다.
// 2명씩 태울 수 있으면 2명씩 태우는 것이 좋다.
// 2명씩 태우는 기준을 어떻게 잡아야 구명보트의 최솟값을 구할 수 있을까
int solution(vector<int> people, int limit) {
    int answer = 0;
    sort(people.begin() , people.end());
    int i = 0 , j = people.size()-1;
    while(i < j){
        if(people[i]+people[j] <= limit){
            i++;
            j--;
        }else {
            j--;
        }
        answer++;
    }
    if(i==j) answer++;
    return answer;
}
```
