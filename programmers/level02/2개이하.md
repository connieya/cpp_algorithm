# 프로그래머스 Lv.2  2개 이하로 다른 비트

## solution 1

```c++
#include <bits/stdc++.h>

typedef long long ll;
using namespace std;

vector<long long> solution(vector<ll> numbers) {
    vector<ll> answer;
    for(auto num : numbers){
        if(num % 2 == 0){
            answer.push_back(num+1);
            continue;
        }
        ll bit = 1;
        while(1){
            if( (bit & num) == 0) break;
            bit *= 2;
        }
        
        answer.push_back(num+bit/2);
    }
    return answer;
}
```
