# 프로그래머스 Lv.2 다음 큰 숫자

## solution 1

```c
#include <bits/stdc++.h>


using namespace std;

int k = 0;
void cnt(int n){
    if(n == 0) return ;
    k += (n%2 == 1);
    cnt(n/2);
}

int solution(int n) {
    cnt(n);
    int t = k;
    for(int i=n+1; ; i++){
        k = 0;
        cnt(i);
        if(k == t) return i;
    }
    return -1;
}
```

## solution 4

```c
#include <bits/stdc++.h>

using namespace std;

int count_bit_on(int num){
    int cnt = 0;
    while(num){
        cnt += (num & 1);
        num >>= 1;
    }
    return cnt;
}


int solution(int n) {
    int num = count_bit_on(n);
    ++n;
    while(num != count_bit_on(n)) {
        n++;
    }
    return n;
}
```
