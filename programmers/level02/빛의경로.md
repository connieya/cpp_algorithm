# 프로그래머스 Lv.2 빛의 경로 사이클

## solution 1

```c++
#include <bits/stdc++.h>

using namespace std;

int row,col;
bool check[500][500][4]={false,};
vector<string> mp;
int dx[4]={0,1,0,-1};
int dy[4]={-1,0,1,0};

// 경로가 끝났다는 것을 알기 위한 방문 체크와
// 이미 존재하는 사이클인지 구분을 어떻게 해야 할 것인가
int solve(int y, int x ,int dir , int len) {
    if(check[y][x][dir]){
        return len;
    }
    check[y][x][dir]=true;
    char c = mp[y][x];
    int ny,nx,nd=dir;
    if(c=='L')  nd = (dir+3)%4;
    else if(c=='R') nd = (dir+1)%4;
    ny = y+dy[nd];
    nx = x+dx[nd];
    if(ny<0) ny+=row;
    else if(ny>=row) ny-=row;
    if(nx<0) nx+=col;
    else if(nx>=col) nx-=col;
    return solve(ny,nx,nd,len+1);
}

vector<int> solution(vector<string> grid) {
    row = grid.size();
    col = grid[0].size();
    mp = grid;
    vector<int> answer;
    for(int i=0; i<row; i++){
        for(int j=0; j<col; j++) {
            for(int k=0; k<4; k++){
                int res = solve(i,j,k,0);
                if(res) answer.push_back(res);
            }
            
        }
    }
    sort(answer.begin() , answer.end());
    return answer;
}
```
