# 프로그래머스 Lv.2 예상 대진표

## solution 1

```c++
#include <bits/stdc++.h>

using namespace std;

int solution(int n, int a, int b)
{
    int answer = 0;
    while(n){
        answer++;
        if( (a+1)/2 == (b+1)/2 ) return answer;
        a = (a+1)/2;
        b = (b+1)/2;
        n/=2;
    }
    return answer;
}
```

## solution 2

```c++
#include <bits/stdc++.h>

using namespace std;

int solution(int n, int a, int b)
{
    a--;
    b--;
    int answer=0;
    while(a!=b){
        a=a/2;
        b=b/2;
        answer++;
    }
    return answer;
}
```
