# 프로그래머스 Lv.2 후보키


## Solution 1

```c
#include <bits/stdc++.h>

using namespace std;

int countBits(int n) {
    int ret = 0;
    while(n){
        ret += n&1;
        n = n >> 1;
    }
    return ret;
}

bool cmp(int a, int b){
    return countBits(a) > countBits(b);
}

int solution(vector<vector<string>> relation) {
    int answer = 0;
    int len = relation.size();
    int n = relation[0].size();
    vector<int> candidates;
    for(int bit=1; bit < 1<<n; bit++){
        set<string> S;
        for(int i=0; i<len; i++){
            auto r = relation[i];
            string tmp = "";
            for(int j=0; j< r.size(); j++){
                if(bit & 1 << j) tmp += r[j];
            }
            S.insert(tmp);
        }
       if(S.size() == len){
           candidates.push_back(bit);
       }
    }
    sort(candidates.begin() , candidates.end() ,cmp);
    
    while(!candidates.empty()){
        int n = candidates.back();
        candidates.pop_back();
        answer++;
        for(vector<int>:: iterator it = candidates.begin(); it != candidates.end();) {
            if((n & *it) == n) it = candidates.erase(it);
            else it++;
        }
    }
    return answer;
}
```

## Solution 2

```c
#include <bits/stdc++.h>

using namespace std;

bool check(vector<int> &v , int now)
{
    for(int i=0; i < v.size(); i++) {
        if( (v[i] & now) == v[i]) return false;
    }
    return true;
}


int solution(vector<vector<string>> relation) {
    int answer = 0;
    int len = relation.size();
    int n = relation[0].size();
    vector<int> candidates;
    for(int bit=1; bit < 1<<n; bit++){
        set<string> S;
        for(int i=0; i<len; i++){
            auto r = relation[i];
            string tmp = "";
            for(int j=0; j< n; j++){
                if(bit & 1 << j) tmp += r[j];
            }
            S.insert(tmp);
        }
       if(S.size() == len && check(candidates , bit)){
           candidates.push_back(bit);
       }
    }
  
    return candidates.size();
}
```
