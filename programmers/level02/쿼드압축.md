# 프로그래머스 Lv.2 쿼드압축 후 개수 세기

## solution 1

```c++
#include <bits/stdc++.h>

using namespace std;

vector<vector<int>> board;

int zero,one;

void dfs(int x,int y, int n) {
    int num = board[x][y];
    for(int i=x; i< x+n; i++){
        for(int j=y; j<y+n; j++){
            if(num != board[i][j]){
                num = -1;
            }
        }
        if(num == -1) break;
    }
    if(num == 1){
        one++;
    }else if(num == 0){
        zero++;
    }else {
        dfs(x,y,n/2);
        dfs(x+n/2,y,n/2);
        dfs(x,y+n/2,n/2);
        dfs(x+n/2,y+n/2,n/2);
    }
}

vector<int> solution(vector<vector<int>> arr) {
    board = arr;
    int n = arr.size();
    dfs(0,0,n);
    
    return {zero,one};
}
```
