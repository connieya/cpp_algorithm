# 프로그래머스 Lv.2 줄서는 방법

## solution 1

```c
#include <bits/stdc++.h>

typedef long long ll;

using namespace std;

bool check[21];

vector<int> solution(int n, ll k) {
    vector<int> answer(n);
    ll sz[21]= {1,1,};
    for(int i=2; i<=20; i++){
        sz[i] = sz[i-1]*i;
    }
    
    for(int i=0; i<n; i++){
        for(int j=1; j<=n; j++){
            if(check[j]) continue;
            if(sz[n- (i+1)] < k){
                k -= sz[n-(i+1)];
            }else{
                answer[i] = j;
                check[j] = true;
                break;
            }
        }
    }
    return answer;
}
```
