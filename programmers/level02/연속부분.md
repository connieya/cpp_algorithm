# 프로그래머스 Lv.2 연속 부분 수열 합의 개수

## solution 1

```c++
#include <bits/stdc++.h>
using namespace std;

int solution(vector<int> elements) {
    int len = elements.size();
    set<int> s;
    for(int i=0; i<len; i++){
        int sum = elements[i];
        s.insert(sum);
        for(int j=i+1; j<len; j++){
            sum += elements[j];
            s.insert(sum);
        }
        for(int k=0; k<i-1; k++){
            sum += elements[k];
            s.insert(sum);
        }
        
    }
    return s.size();
}
```
