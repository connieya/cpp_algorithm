# 프로그래머스 Lv.2 괄호변환

## solution 1

```c
#include <bits/stdc++.h>

using namespace std;

string solution(string p) {
    string answer = "";
    if(p.empty()) return answer;
    int cnt = 0 , idx = 0;
    for(; idx < p.size(); idx++ ){
        if(p[idx] == '(') cnt++;
        else cnt--;
        if(!cnt) break;
    }
    idx++;
    string u = p.substr(0,idx) , v;
    if(idx < p.size()) v = p.substr(idx, p.size());
    
    bool flag = 1;
    for(int i=0; i < u.size() && flag; i++) {
        if(p[i] == '(') cnt++;
        else cnt--;
        if(cnt < 0) flag = 0;
    }
    
    if(flag) return u + solution(v);
    else {
        u.erase(0,1);
        u.pop_back();
        for(auto &c : u) c = (c== '(' ? ')' : '(');
        return string(1, '(') + solution(v) + string(1,')') + u;
    }
    
    return answer;
}
```

## solution 2


```c
#include <bits/stdc++.h>

using namespace std;

bool check(string str){
    int cnt =0;
    for(int i=0; i<str.size(); i++){
        cnt += str[i] == '(' ? 1 : -1;
        if(cnt < 0) return false;
    }
    return true;
}

string solution(string p) {
    string answer = "";
    if(p.empty()) return answer;
    if(check(p)) return p;
    int cnt = 0 , idx = 0;
    for(idx = 0; idx < p.size(); idx++ ){
        if(p[idx] == '(') cnt++;
        else cnt--;
        if(!cnt) break;
    }
    string u = p.substr(0,idx+1);
    string v = p.substr(idx+1); 
    if(check(u)) return u + solution(v);
    else {
        u.erase(0,1);
        u.pop_back();
        for(auto &c : u) c = (c== '(' ? ')' : '(');
        return string(1, '(') + solution(v) + string(1,')') + u;
    }
    
    return answer;
}
```
