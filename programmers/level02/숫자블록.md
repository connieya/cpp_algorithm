# 프로그래머스 Lv.2 숫자 블록

## solution 1

```c++
#include <bits/stdc++.h>

typedef long long ll;
using namespace std;

int solve(ll n){
    if(n==2) return 1;
    for(ll i=2; i*i <=n; i++){
        if(n%i == 0 && n/i <= 10000000) return n/i;
    }
    return 1;
}

vector<int> solution(ll begin, ll end) {
    vector<int> answer;
    if(begin == 1){
        answer.push_back(0);
        begin++;
    }
    for(ll i = begin; i <= end; i++){
        answer.push_back(solve(i));
    }
    return answer;
}
```
