# 프로그래머스 Lv.2  숫자의 표현

## solution 1

```c
#include <bits/stdc++.h>

using namespace std;

int solution(int n) {
    int answer = 1;
    int lt = 1 , rt =1;
    int sum = 0;
    while(rt <= n){
        if(sum == n) answer++;
        if(sum > n) {
            sum -= lt++;
        }else {
            sum += rt++;
        }
    }
    return answer;
}
```

## solution 2 java


```c
class Solution {
    public int solution(int n) {
        int sum = 0;
        int count = 0;
        int j = 1;
        for (int i = 1; i <= n; i++) {
            sum += i;
            while (sum >= n) {
               if (sum==n) count++;
               sum -=j;
               j++;
            }
        }
        return count;
    }
}
```
