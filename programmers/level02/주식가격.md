# 프로그래머스 Lv.2 주식 가격

## solution 1

```c++
#include <bits/stdc++.h>

using namespace std;

vector<int> solution(vector<int> prices) {
    vector<int> answer(prices.size());
    stack<int> s;
    for(int i=0; i<prices.size(); i++) {
        while(!s.empty() && prices[s.top()] > prices[i]){
            answer[s.top()] = i-s.top();
            s.pop();
        }
        s.push(i);
    }
    while(!s.empty()){
        int idx = s.top();
        s.pop();
        answer[idx] = prices.size()-1-idx;
    }
    return answer;
}
```
