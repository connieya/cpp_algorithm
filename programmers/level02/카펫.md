# 프로그래머스 Lv.2 카펫


## solution 1

```c
#include <bits/stdc++.h>

using namespace std;

vector<int> solution(int brown, int yellow) {
    vector<int> answer;
    if(yellow % 2) {
        answer.push_back(yellow+2);
        answer.push_back(3);
    }else {
        int idx = 1;
        while(idx*idx <= yellow){
            if(yellow % idx == 0){
                int v = yellow / idx;
                if(brown == (v+2)*2 +2*idx){
                    answer.push_back(v+2);
                    answer.push_back(idx+2);
                    break;
                }
            }
            idx++;
        }
    }
    return answer;
}
```
