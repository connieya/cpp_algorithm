# 프로그래머스 Lv.2 카펫

Yellow 의 약수를 구한 뒤 해당 조건에 맞는지 확인

## solution 1


## 틀린 코드

```c
#include <bits/stdc++.h>

using namespace std;

vector<int> solution(int brown, int yellow) {
    vector<int> answer;
    if(yellow % 2) {
        answer.push_back(yellow+2);
        answer.push_back(3);
    }else {
        int idx = 1;
        while(idx*idx <= yellow){
            if(yellow % idx == 0){
                int v = yellow / idx;
                if(brown == (v+2)*2 +2*idx){
                    answer.push_back(v+2);
                    answer.push_back(idx+2);
                    break;
                }
            }
            idx++;
        }
    }
    return answer;
}
```
