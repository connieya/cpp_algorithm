# 프로그래머스 Lv.2 전력망을 둘로 나누기

## Solution 1

### DFS  wire의 크기만큼


```c
#include <bits/stdc++.h>

using namespace std;

vector<int> graph[101];
bool checked[101];

void dfs(int now ,int prev , int target) {
    checked[now] = true;
    for(int nxt : graph[now]){
        if( (prev == nxt) || (nxt == target)) continue;
        dfs(nxt ,now, target); 
    }
}
void init(int n) {
    for(int i=1; i<=n; i++){
        checked[i] = false;
    }
}

int calc(int n){
    int cnt = 0;
    for(int i=1; i<=n; i++){
        cnt += checked[i] ? 1 : -1;
    }
    return cnt < 0 ? -cnt : cnt;
}

int solution(int n, vector<vector<int>> wires) {
    int answer = 100;
    for(auto wire : wires){
        graph[wire[0]].push_back(wire[1]);
        graph[wire[1]].push_back(wire[0]);
    }
    for(auto wire : wires){
        int a = wire[0] , b = wire[1];
        dfs(a,-1,b);
        answer = min(answer, calc(n));
        init(n);
        
    }
    return answer;
}
```

## Solution 2

```c
#include <bits/stdc++.h>

using namespace std;

vector<int> graph[101];
int sz[101];

void dfs(int start ,int prev){
    sz[start] = 1;
    for(int next : graph[start]) {
        if(next == prev) continue;
        dfs(next , start);
        sz[start] += sz[next];
    }
}


int solution(int n, vector<vector<int>> wires) {
    int answer = INT_MAX;
    for(int i=0; i< n-1; i++){
        graph[wires[i][0]].push_back(wires[i][1]);
        graph[wires[i][1]].push_back(wires[i][0]);
    }
    dfs(1,-1);
    for(int i=1; i<=n; i++){
        for(int j : graph[i]){
            int l = sz[j];
            int r = n-l;
            answer = min(answer , abs(l-r));
        }
    }
    return answer;
}
```

## solution 3

```c++
#include <bits/stdc++.h>

using namespace std;

vector<int> graph[101];
int sz[101];

void dfs(int now ,int prev ) {
    sz[now] =1;
    for(int nxt : graph[now]){
        if(prev == nxt) continue;
        dfs(nxt ,now); 
        sz[now] += sz[nxt];
    }
}


int solution(int n, vector<vector<int>> wires) {
    int answer = 100;
    for(auto wire : wires){
        graph[wire[0]].push_back(wire[1]);
        graph[wire[1]].push_back(wire[0]);
    }
    dfs(1,-1);
    for(auto wire : wires){
        int a = wire[0] , b = wire[1];
        int l = sz[b];
        int r = n-l;
        answer = min( {answer, abs(l-r) , abs(sz[a]-(n-sz[a]))});
    }
    return answer;
}
```
