# 프로그래머스 Lv.2 올바른 괄호

## solution 1

```c
#include<bits/stdc++.h>

using namespace std;

bool check(string s){
    int cnt = 0;
    for(auto ch : s){
        if(ch == '('){
            cnt++;
        }else {
            if(cnt == 0) return false;
            cnt--;
        }
    }
    return cnt == 0;
}

bool solution(string s)
{
    return check(s);
}
```
