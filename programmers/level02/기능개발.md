# 프로그래머스 Lv.2 기능 개발


## solution 1

```c++
#include <bits/stdc++.h>


using namespace std;

vector<int> solution(vector<int> progresses, vector<int> speeds) {
    vector<int> answer;
    int len = progresses.size();
    int day = 0;
    for(int i=0; i<len; i++) {
        int tmp = (100-progresses[i]) % speeds[i];
        if(tmp == 0){
            tmp = (100-progresses[i]) / speeds[i];
        }else {
            tmp = (100-progresses[i]) / speeds[i] +1;
        }
        if(tmp > day){
            answer.push_back(1);
        }else {
            ++answer.back();
        }
        if(tmp > day) {
            day = tmp;
        }
    }
  
    return answer;
}
```

## solution 2



```c++
#include <bits/stdc++.h>


using namespace std;

vector<int> solution(vector<int> progresses, vector<int> speeds) {
    vector<int> answer;
    queue<int> q;
    for(int i=0; i< speeds.size(); i++) {
        int remain = 100-progresses[i];
        int day = remain % speeds[i] ? remain / speeds[i]+1 : remain / speeds[i];
        q.push(day);
    }
    int cnt = 1;
    while(!q.empty()){
        int cur = q.front();
        q.pop();
        while(!q.empty() && cur >= q.front()){
            cnt++;
            q.pop();
        }
        answer.push_back(cnt);
        cnt = 1;
    }
    
    return answer;
}
```
