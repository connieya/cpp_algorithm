# 프로그래머스 Lv.2 스킬 트리

## solution 1

```c++
#include <bits/stdc++.h>

using namespace std;

int solution(string skill, vector<string> skill_trees) {
    int answer = 0;
    int number [27] = {0};
    for(auto s : skill){
        number[s-'A']++;
    }
    for(auto tree : skill_trees){
        int idx = 0;
        bool flag = true;
        for(auto t : tree){
            if(!number[t-'A']) continue;
            if(skill[idx] == t){
                idx++;
            }else {
                flag = false;
                break;
            } 
        }
        answer += flag; 
    }
    return answer;
}
```
