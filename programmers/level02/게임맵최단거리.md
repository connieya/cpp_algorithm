# 프로그래머스 Lv.2 게임 맵 최단거리

## solution 1

```c
#include<bits/stdc++.h>

using namespace std;

int dp[101][101];
int n,m;

int dx[] = {-1,0,1,0};
int dy[] = {0,1,0,-1};

void bfs(vector<vector<int>> &maps){
    dp[0][0] = 1;
    queue<pair<int,int>> Q;
    Q.push({0,0});
    while(!Q.empty()){
        int x = Q.front().first;
        int y = Q.front().second;
        Q.pop();
        for(int i=0; i<4; i++){
            int nx = x +dx[i];
            int ny = y +dy[i];
            if(nx < 0 || nx >=n || ny < 0 || ny >= m) continue;
            if(dp[nx][ny] == -1 && maps[nx][ny]){
                dp[nx][ny] = dp[x][y] +1;
                Q.push({nx,ny});
            }
        }
        
    }
}

int solution(vector<vector<int> > maps)
{
    n = maps.size();
    m = maps[0].size();
    memset(dp,-1, sizeof(dp));
    bfs(maps);
    return dp[n-1][m-1];
}
```
