# 프로그래머스 Lv.2 오픈 채팅방


## solution 1

```c++
#include "bits/stdc++.h"

using namespace std;

vector<string> solution(vector<string> record) {
    vector<string> answer;
    unordered_map<string, string> user_id;
    for(string re : record) {
        stringstream s(re);
        string command, id , nickname;
        s >> command >> id >> nickname;
        if(command != "Leave") {
            user_id[id] = nickname;
        }
    }
    for(string re : record) {
        stringstream s(re);
        string command, id , nickname;
        s >> command >> id >> nickname;
        string res;
        if(command == "Change") {
           continue;
        }
        string n = user_id[id];
        if(command == "Enter"){
            res = n+"님이 들어왔습니다.";
        }else if(command == "Leave") {
            res = n+"님이 나갔습니다.";
        }
        answer.push_back(res);
    }
    
    
    return answer;
}
```
## solution 2

```c++
#include <bits/stdc++.h>

using namespace std;

vector<string> solution(vector<string> record) {
    vector<string> answer;
    unordered_map<string,string> mp;
    string a,b,c;
    for(auto r : record){
        stringstream iss(r);
        iss >> a >> b;
        if(a == "Enter" || a == "Change"){
            iss >> c;
            mp[b] = c;
        }
    }
    for(auto r : record){
        stringstream iss(r);
        iss >> a  >> b;
        if(a == "Change") continue;
        if(a=="Enter"){
            answer.push_back(mp[b]+"님이 들어왔습니다.");
        }else{
            answer.push_back(mp[b]+"님이 나갔습니다.");
        }
    }
    return answer;
}
```
