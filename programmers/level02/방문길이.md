# 프로그래머스 Lv.2 방문 길이

## solution 1

```c++
#include <bits/stdc++.h>
using namespace std;

int dx [] = {1,0,-1,0};
int dy [] = {0,1,0,-1};


bool oop(int x, int y){
    return x < -5 || x > 5 || y < -5 || y > 5;
}


int solution(string dirs) {
    int answer = 0;
    unordered_map<char,int> mp;
    unordered_map<string,bool> check;
    mp['U'] = 0 , mp['R'] = 1 ,mp['D'] = 2 ,mp['L'] = 3;
    int x = 0 ,y = 0 ,nx ,ny;
    for(auto dir : dirs){
        int idx = mp[dir];
        if(oop(x+dx[idx], y+dy[idx])) continue;
        nx = x+dx[idx];
        ny = y+dy[idx];
        string str1 = to_string(x)+to_string(y)+to_string(nx)+to_string(ny);
        string str2 = to_string(nx)+to_string(ny)+to_string(x)+to_string(y);
        if(!check[str1] && !check[str2]){
            answer++;
            check[str1] = true;
            check[str2] = true;
        }
        x =nx;
        y =ny;
    }
    return answer;
}
```
