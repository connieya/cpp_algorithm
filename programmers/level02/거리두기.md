# 프로그래머스 Lv.2 거리두기 확인하기

## solution 1

```c++
#include <bits/stdc++.h>

using namespace std;

int dx[] = {-1,0,1,0};
int dy[] = {0,1,0,-1};

bool solve(vector<string> &place){
    queue<pair<int,int>> q;
    int number[5][5] = {0,};
    int num = 0;
    
    for(int i=0; i<5; i++) {
        for(int j=0; j< 5; j++){
            if(place[i][j] == 'P'){
                num++;
                for(int k = 0; k< 4; k++){
                    int nx = dx[k]+i;
                    int ny = dy[k]+j;
                    if(nx < 0 || nx >=5 || ny < 0 || ny >= 5) continue;
                    if(place[nx][ny] == 'X') continue;
                    if(place[nx][ny] == 'P') return false;
                    number[i][j] = num;
                    number[nx][ny] = num;
                    q.push({nx,ny});   
                }
            }
        }
    }
    
      
        
        while(!q.empty()){
            int x = q.front().first;
            int y = q.front().second;
            q.pop();
            for(int i=0; i<4; i++){
                int nx = dx[i]+x;
                int ny = dy[i]+y;
                if(nx < 0 || nx >= 5 || ny < 0 || ny >= 5) continue;
                if(place[nx][ny] == 'P' && number[nx][ny] != number[x][y]) return false;
            }
        }
    
    return true;
}

vector<int> solution(vector<vector<string>> places) {
    vector<int> answer;
    for(auto &place : places){
       answer.push_back(solve(place));
    }
    return answer;
}
```

## solution 2

```c++
#include <bits/stdc++.h>

using namespace std;

int dx[] = {-1,0,1,0};
int dy[] = {0,1,0,-1};

bool solve(vector<string> &place){
    vector<vector<int>> check(5, vector<int>(5,false));
    
    for(int i=0; i<5; i++) {
        for(int j=0; j< 5; j++){
            if(place[i][j] == 'P'){
                for(int k = 0; k< 4; k++){
                    int nx = dx[k]+i;
                    int ny = dy[k]+j;
                    if(nx < 0 || nx >=5 || ny < 0 || ny >= 5) continue;
                    if(place[nx][ny] == 'P') return false;
                    if(place[nx][ny] == 'O'){
                        if(check[nx][ny]) return false;
                        check[nx][ny] = true;
                    }
                }
            }
        }
    }
      return true;
}

vector<int> solution(vector<vector<string>> places) {
    vector<int> answer;
    for(auto &place : places){
       answer.push_back(solve(place));
    }
    return answer;
}
```
