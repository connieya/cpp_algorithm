# 프로그래머스 Lv.2 행렬 테두리 회전하기

## solution 1

```c
#include <bits/stdc++.h>

using namespace std;
int arr[105][105];


int rotate(int x1 , int y1 ,int x2 ,int y2) {
    int tmp = arr[x1][y1];
    int mx = INT_MAX;
    for(int i = x1; i < x2; i++){
        mx = min(mx , arr[i][y1]);
        arr[i][y1] = arr[i+1][y1];
    }
    for(int i = y1; i < y2; i++){
        mx = min(mx , arr[x2][i]);
        arr[x2][i] = arr[x2][i+1];
    }
    for(int i=x2; i > x1; i--) {
        mx = min(mx , arr[i][y2]);
        arr[i][y2] = arr[i-1][y2];
    }
    for(int i=y2; i > y1; i--){
        mx = min(mx , arr[x1][i]);
        arr[x1][i] = arr[x1][i-1];
    }
    arr[x1][y1+1] = tmp;
    return mx;
}


vector<int> solution(int rows, int columns, vector<vector<int>> queries) {
    vector<int> answer;
    int value = 1;
    for(int i=1; i<=rows; i++){
        for(int j=1; j<=columns; j++) {
            arr[i][j] = value++;
        }
    }
    
    for(auto query : queries){
        int x1 = query[0] , y1 = query[1] , x2 = query[2] , y2 = query[3];
        answer.push_back(rotate(x1,y1,x2,y2));
        
    }
    return answer;
}
```
