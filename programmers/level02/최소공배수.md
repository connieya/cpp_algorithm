# 프로그래머스 Lv.2 N개의 최소 공배수

## solution 1

```c
#include <bits/stdc++.h>

using namespace std;

int gcd(int a,int b){
    if(b == 0) return a;
    return gcd(b,a%b);
}

int lcd (int a , int b){
    return a*b / gcd(a,b);
}

int solution(vector<int> arr) {
    int answer = arr[0];
    for(int i=1; i< arr.size(); i++){
        answer = lcd(arr[i],answer);
    }
    
    return answer;
}
```
