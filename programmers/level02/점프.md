# 프로그래머스 Lv.2 점프와 순간이동

## solution 1

```c
#include <bits/stdc++.h>

using namespace std;

int solution(int n)
{
    int ans = 0;
    while(n) {
        if(n&1) ans++;
        n /= 2;
    }
    return ans;
}
```

## solution 2

```c
#include <bits/stdc++.h>

using namespace std;

int solution(int n)
{
    int ans = 0;
    for(int i=0; i<31; i++){
        if(n & 1<<i) ans++;
    }
    return ans;
}
```

## solution 3

```c
#include <bits/stdc++.h>

using namespace std;

int solution(int n)
{
    return __builtin_popcount(n);
}
```
