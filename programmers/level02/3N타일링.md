# 프로그래머스 Lv.2 3 x n 타일링

## solution 1

```c++
#include <bits/stdc++.h>
#define MOD 1000000007

typedef long long ll;
using namespace std;

ll dp[5001];

int solution(int n) {
    int answer = 0;
    dp[1] = 2;
    dp[2] = 3;
    for(int i=3; i<=n; i++) {
        if(i%2){
            dp[i] = dp[i-1]*2 +dp[i-2];
        }else {
            dp[i] = dp[i-1]+dp[i-2];
        }
        dp[i] %= MOD;
    }
    return dp[n];
}
```
