# 프로그래머스 Lv.2 이진 변환 반복하기

## solution 1

```c++
#include <bits/stdc++.h>

using namespace std;

vector<int> solution(string s) {
    vector<int> answer;
    int remove = 0 , cnt =0 , len = 0;
    string tmp = "";
    while(s !="1"){
        for(auto t : s){
            if(t == '0'){
                remove++;
            }else {
               len++;
            }
        }
        string tmp = "";
        while(len){
            tmp += to_string(len%2);
            len /=2;
        }
        reverse(tmp.begin(), tmp.end());
        s = tmp;
        cnt++;
    }
    return {cnt, remove};
}
```
