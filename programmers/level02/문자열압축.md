#include <bits/stdc++.h>


using namespace std;

vector<string> convert(string s , int sz){
    vector<string> arr;
    for(int i=0; i<s.length(); i+=sz){
        arr.push_back(s.substr(i,sz));
    }
    return arr;
}

int compress(vector<string> &v , int sz){
    int cnt = 1;
    string tmp = "";
    string prev = "";
    for(auto s : v){
        if(prev == s){
            cnt++;
        }else {
            if(cnt > 1){
                tmp += to_string(cnt)+prev;
            }else {
                tmp += prev;
            }
            cnt = 1;
        }
        prev = s;
    }
    if(cnt > 1){
        tmp += to_string(cnt)+prev;
    }else {
        tmp += prev;
    }
    return tmp.length();
}

int solution(string s) {
    int answer = s.length();
    for(int i=1; i<= s.length()/2; i++) {
        vector<string> v = convert(s,i);
        int res = compress(v,i);
        if(answer > res) answer = res;   
    }
    return answer;
}
