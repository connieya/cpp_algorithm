# 프로그래머스 Lv.2 위장


## solution 1

```c
#include <bits/stdc++.h>

using namespace std;

int solution(vector<vector<string>> clothes) {
    int answer = 1;
    unordered_map<string,int> mp;
    for(auto clothe : clothes){
        mp[clothe[1]]++;
    }
    for(auto m : mp){
        answer *= (m.second+1);
    }
    return answer-1;
}
```
