# 프로그래머스 Lv.2 영어 끝말잇기

## solution 1

```c
#include <bits/stdc++.h>

using namespace std;

string solution(string number, int k) {
    string answer = "";
    stack<char> s;
    for(auto n : number) {
        while(s.size() && s.top() < n && k){
            s.pop();
            k--;
        }
        s.push(n);
    }
    
    while(k){
        s.pop();
        k--;
    }
    
    while(s.size()){
        answer += s.top();
        s.pop();
    }
    reverse(answer.begin(), answer.end());
    return answer;
}
```
