# 프로그래머스 Lv.2 프렌즈 블록

## solution 1

```c++
#include <bits/stdc++.h>

using namespace std;

bool check[35][35];
bool isValid(int x, int y , vector<string> board){
    return (board[x][y] == board[x+1][y]) && (board[x+1][y] == board[x+1][y+1]) 
        && (board[x+1][y+1] == board[x][y+1]);
}

// 2x2 블록을 어떻게 구분하는가?
// 블록이 사라진 뒤 아래로 떨어지는 로직은 어떻게 구현하는가?
int solution(int m, int n, vector<string> board) {
    int answer = 0;
    int cnt = 0;
    while(1){
        bool flag = true;
        for(int i=0; i<m; i++){
            for(int j=0; j<n; j++){
                if(i == m-1 || j == n-1 ) continue;
                if(isValid(i,j,board) && board[i][j] != '.'){
                    cout << i << ' ' << j << '\n';
                    flag =false;
                    check[i][j] = check[i+1][j] = true;
                    check[i][j+1] = true;
                    check[i+1][j+1] = true;
                }
            }
        }
        
        for(int i=0; i<m; i++){
            for(int j=0; j<n; j++){
                if(check[i][j]){
                    answer++;
                    board[i][j] = '.';
                }
                check[i][j] = false;
            }
        }
        
        
        for(int j=0; j<n; j++){
            for(int i=m-1; i>0; i--){
                if(board[i][j] == '.'){
                    int x = i-1;
                    while(board[x][j] == '.' && x > 0){
                        x--;
                    }
                    board[i][j] = board[x][j];
                    board[x][j] = '.';
                }
            }
        }
        
        
        if(flag) break;
    }
    return answer;
}
```
