# 프로그래머스 Lv.2 프린터

## solution 1

```c
#include <bits/stdc++.h>

using namespace std;

int solution(vector<int> priorities, int location) {
    int answer = 1;
    queue<pair<int,int>> Q;
    for(int i=0; i < priorities.size(); i++) {
        Q.push({i,priorities[i]});
    }
    while(!Q.empty()){
        int len = Q.size();
        int init = Q.front().second;
        int idx = Q.front().first;
        bool flag = true;
        for(int i=0; i<len; i++) {
            int now = Q.front().second;
            int idx = Q.front().first;
            Q.pop();
            if(now > init) flag =false;
            Q.push({idx,now});
        }
        if(flag) {
            if(idx == location) return answer;
            Q.pop();
            answer++;
        }else {
            auto cur = Q.front();
            Q.pop();
            Q.push(cur);
        }
    }
    return answer;
}
```

## solution 2

```c
#include <bits/stdc++.h>

using namespace std;

int solution(vector<int> priorities, int location) {
    queue<int> q;
    vector<int> v;
    for(int i=0; i<priorities.size(); i++){
        q.push(i);
    }
    while(!q.empty()){
        int now = q.front();
        q.pop();
        if( priorities[now] == *max_element( priorities.begin(), priorities.end() )) {
            priorities[now] = 0;
            v.push_back(now);
        }else {
            q.push(now);
        }
    }
    
    for(int i=0; i<v.size(); i++){
        if(v[i] == location) return i+1;
    }
         
}
```

## solution 3

### 큐 안쓰기 풀기

```c
#include <bits/stdc++.h>

using namespace std;

int solution(vector<int> priorities, int location) {
    int answer = 0;
    int mx = *max_element(priorities.begin() , priorities.end());
    while(1) {
        for(int i=0; i<priorities.size(); i++){
            if(priorities[i] == mx) {
                answer++;
                priorities[i] = 0;
                if(i == location) return answer;
            }
            mx = *max_element(priorities.begin() , priorities.end());
        }
    }
}
```
