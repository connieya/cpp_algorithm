# 프로그래머스 Lv.2 짝지어 제거하기

## solution 1

```c
#include <bits/stdc++.h>

using namespace std;

int solution(string s)
{
    stack<char> S;
    for(char ch: s){
        if(!S.empty() && S.top() ==  ch) {
            S.pop();
        }else {
            S.push(ch);
        }
    }

    return S.empty() ? 1 : 0;
}
```

## solution 2

```c
#include <bits/stdc++.h>

using namespace std;
int solution(string s)
{
    stack<char> st;
    for(auto ch : s) {
        if(!st.empty() && st.top() == ch) {
            st.pop();
            continue;
        }
        st.push(ch);
    }
    return st.empty() ? 1 : 0;
}
```
