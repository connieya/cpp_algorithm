# 프로그래머스 Lv.2 피로도

## solution 1

```c
#include <bits/stdc++.h>

using namespace std;

int solution(int k, vector<vector<int>> dungeons) {
    int answer = 0;
    int len = dungeons.size();
    vector<int> tmp;
    for(int i=0; i< len; i++){
        tmp.push_back(i);
    }
    
    do {
        int cnt = 0;
        int t = k;
        for(int i=0; i< len; i++){
            int idx = tmp[i];
            if(t >= dungeons[idx][0]) {
                cnt++;
                t -= dungeons[idx][1];
            }
        }
        if(cnt > answer){
            answer = cnt;
        }
        
    }while(next_permutation(tmp.begin(), tmp.end() ) );
    
    return answer;
}
```
