# 프로그래머스 Lv.2 [1차] 뉴스 클러스터링

## solution 1

```c
#include <bits/stdc++.h>

using namespace std;

int solution(string str1, string str2) {
    int answer = 0;
    vector<string> v1;
    vector<string> v2;
    for(auto &str : str1){
        if(str >= 97 && str <= 122) str -= 32;
    }
    for(auto &str : str2){
        if(str >= 97 && str <= 122) str -= 32;
    }
    
    for(int i=0; i<str1.size()-1; i++){
        string tmp;
        if(isalpha(str1[i]) && isalpha(str1[i+1])){
            tmp += str1[i];
            tmp += str1[i+1];
            v1.push_back(tmp);
        }
    }
    
    for(int i=0; i<str2.size()-1; i++){
        string tmp;
        if(isalpha(str2[i]) && isalpha(str2[i+1])){
            tmp += str2[i];
            tmp += str2[i+1];
            v2.push_back(tmp);
        }
    }
    
    sort(v1.begin() , v1.end());
    sort(v2.begin() , v2.end());
    set<string> s;
    for(auto v : v1) s.insert(v);
    for(auto v : v2) s.insert(v);
    double i = 0;
    double u = 0;
    for(auto st : s){
        int str1_cnt = (upper_bound(v1.begin(),v1.end(), st)-v1.begin())-
            (lower_bound(v1.begin(),v1.end(),st)-v1.begin());
        int str2_cnt = (upper_bound(v2.begin(),v2.end(), st)-v2.begin())-
            (lower_bound(v2.begin(),v2.end(),st)-v2.begin());
        
        int t1 = min(str1_cnt,str2_cnt);
        int t2 = max(str1_cnt,str2_cnt);
        i += t1;
        u += t2;
    }
   
    if(!i && !u) return 65536;
    
    answer = (i/u)*65536;
    return answer;
}
```

## solution 2

```c
#include <bits/stdc++.h>

using namespace std;

int arr1[676];
int arr2[676];

int solution(string A, string B) {
   for(int i=0; i<A.size()-1; i++)
        if(isalpha(A[i]) && isalpha(A[i+1])){
            int idx = (A[i]&31) *26 + (A[i+1]&31);
            arr1[idx]++;
        }
    
    for(int i=0; i<B.size()-1; i++)
        if(isalpha(B[i]) && isalpha(B[i+1])){
            int idx = (B[i]&31)*26 + (B[i+1]&31);
            arr2[idx]++;
        }
            
    int a = 0;
    int b = 0;
    
    for(int i=0; i< 676; i++){
        a += min(arr1[i],arr2[i]);
        b += max(arr1[i],arr2[i]);
    }
    return b ? a*65536/b : 65536;
}
```
