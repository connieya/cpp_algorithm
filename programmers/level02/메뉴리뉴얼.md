# 프로그래머스 Lv.2 메뉴 리뉴얼

## solution 1

```c
#include <bits/stdc++.h>


using namespace std;

set<string> v[11];
unordered_map<string ,int> mp;

void dfs(int L , int len , string order , string str , int start){
    if(L == len){
        mp[str]++;
        v[len].insert(str);
        return;
    }
    
    for(int i=start; i< order.size(); i++) {
        dfs(L+1,len,order, str+order[i],i+1);
    }
}

vector<string> solution(vector<string> orders, vector<int> course) {
    vector<string> answer;
    for(auto &order : orders){
        sort(order.begin(),order.end());
        for(int c : course){
            dfs(0,c,order,"",0);
        }   
    }
    
    for(auto c : course) {
        int mx = 2;
        vector<string> tmp;
        for(auto menu : v[c]) {
            if(mp[menu] > mx){
                tmp.clear();
                mx = mp[menu];
                tmp.push_back(menu);
            }else if(mp[menu] == mx){
                tmp.push_back(menu);
            }
        }
        for(auto t : tmp){
            answer.push_back(t);
        }
    }
    sort(answer.begin(),answer.end());
    return answer;
}
```
