# 2022 상반기 카카오 인턴 2번 문제 - 합이 같은 큐 2개


## Solution 1

```c++
#include <bits/stdc++.h>

typedef long long ll;
using namespace std;


ll solve(vector<int> &mn_queue , vector<int> &mx_queue , ll mn_v , ll mx_v  , ll s) {;
    int cnt = 0;
    int idx_mn = 0;
    int idx_mx = 0;
    int len = mx_queue.size();
    while(1) {
        if(mn_v < s) {
            mn_queue.push_back(mx_queue[idx_mx]);
            mn_v += mx_queue[idx_mx++];
        }else if(mn_v > s){
            mx_v += mn_queue[idx_mn];
            mn_v -= mn_queue[idx_mn++];
        }else {
            return cnt;
        }
        cnt++;
        if(idx_mx == len) break;
    }
    while(mn_v > s) {
        mx_v += mn_queue[idx_mn];
        mn_v -= mn_queue[idx_mn++];
        cnt++;
    }
    if(mn_v == s && mx_v == s) return cnt;
    
    return -1;
}


ll solution(vector<int> queue1, vector<int> queue2) {
    int answer = -1;
    ll s = 0;
    ll q1_s = 0;
    ll q2_s = 0;
    int len = queue1.size();
    for(int i=0; i<len; i++){
        q1_s += queue1[i];
        q2_s += queue2[i];
    }
    s =  (q1_s + q2_s) /2;
    if(q1_s == s) return 0;

    vector<int> mn = queue1;
    vector<int> mx = queue2;
    ll mn_v = q1_s;
    ll mx_v = q2_s;
    if(q1_s > q2_s) {
        vector<int> tmp = mn;
        mn = mx;
        mx = tmp;
        ll tmp2 = mn_v;
        mn_v = mx_v;
        mx_v = tmp2;
    }
    return solve(mn, mx , mn_v , mx_v , s);
}
```
