# 프로그래머스 Lv.1 로또의 최고 순위와 최저 순위

## solution 1

```c++
#include <string>
#include <vector>

using namespace std;

vector<int> solution(vector<int> lottos, vector<int> win_nums) {
    int rank[] = {6,6,5,4,3,2,1};
    int arr[46] = {0,};
    int zero_count = 0;
    int count = 0;
    vector<int> answer;
    for(int num : win_nums){
        arr[num]++;
    }
    for(int lotto : lottos) {
        if(lotto ==  0){
            zero_count++;
            continue;
        }
        arr[lotto]++;
        if(arr[lotto] == 2){
            count++;
        }
    }
    answer.push_back(rank[count+zero_count]);
     answer.push_back(rank[count]);
    return answer;
}
```

## solution 2

```c++
#include <bits/stdc++.h>

using namespace std;

vector<int> solution(vector<int> lottos, vector<int> win_nums) {
    int rank[] = {6,6,5,4,3,2,1};
    int cnt = 0;
    int zero = 0;
    for(int l : lottos){
        if(!l) zero++;
        for(int w : win_nums){
            if(l==w){
                cnt++;
                break;
            }
        }
    }
    return {rank[cnt+zero],rank[cnt]};
}
```
