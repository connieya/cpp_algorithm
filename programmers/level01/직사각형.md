# 프로그래머스 Lv.1 최소직사각형

## solution 1

```c++
#include <bits/stdc++.h>
using namespace std;

int solution(vector<vector<int>> sizes) {
    int a = 0 , b = 0;
    for(auto &size : sizes){
        if(size[0] < size[1]){
            swap(size[0],size[1]);
        }
        a = max(a,size[0]);
        b = max(b,size[1]);
    }
    return a*b;
}
```
