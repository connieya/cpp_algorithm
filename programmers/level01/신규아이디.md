# 프로그래머스 Lv.1 신규 아이디 추천

## solution 1

```c++
#include <string>
#include <vector>
#include <iostream>

using namespace std;

string solution(string new_id) {
    string answer = "";
    for(char &ch : new_id) {
        if (ch >= 'A' && ch <= 'Z') {
            ch += 32;
        }
    }
    for(char &ch : new_id) {
        if (ch >= 'a' && ch <= 'z') {
            answer += ch;
        }
        if (ch >= '0' && ch <= '9') {
            answer += ch;
        }
        if (ch == '-' || ch == '_' || ch == '.') {
            answer += ch;
        }
    }
     
    new_id = answer;
    answer.clear();
    for(char &ch : new_id) {
            if(!answer.empty() && answer.back()=='.' && ch == '.'){
                continue;
            }
            answer += ch;
        }
    
    if(answer.front() == '.') answer.erase(answer.begin());
    if(answer.back() == '.') answer.pop_back();
    if(answer.empty()) answer = "a";
    if(answer.length() >= 16) answer = answer.substr(0,15);
    if(answer.back() == '.') answer.pop_back();
    while(answer.length() <=2) {
        answer += answer.back();
      }
    
    
    return answer;
}
```

#include <bits/stdc++.h>

using namespace std;

string solution(string new_id) {
    string answer = "";
    for(auto &n : new_id) n = tolower(n); // 1단계
    for(auto &n : new_id){
        if(isdigit(n) || strchr("-_.", n) || (n>= 'a' && n<='z')){ // 2단계
            answer += n;
        }
    }
    new_id.clear();
    for(auto &a: answer){
        if(!new_id.empty() && new_id.back() =='.' && a == '.') continue;
        new_id += a;
    }
    answer = new_id;
    if(answer.front() == '.'){
        answer = answer.substr(1);
    }
    if(answer.back() == '.'){
        answer.pop_back();
    }
        
    
    if(answer.empty()) answer += 'a';
    if(answer.size() >= 16) answer = answer.substr(0,15);
    if(answer.back() == '.') answer.pop_back();
    while(answer.size() <= 2){
        answer += answer.back();
    }
    
    
    return answer;
}
