# 프로그래머스 Lv.1 [1차] 비밀지도

## solution 1

```c
#include <bits/stdc++.h>

using namespace std;

vector<string> solution(int n, vector<int> arr1, vector<int> arr2) {
    vector<string> answer;
    for(int a : arr1) {
        string tmp;
        while(a){
            tmp += (a%2+'0');
            a/=2;
        }
        reverse(tmp.begin() , tmp.end());
        while(tmp.size() != n){
            tmp = '0'+tmp;
        }
        string str;
        for(auto t : tmp){
            str += t =='0' ? ' ' : '#';
        }
        answer.push_back(str);
    }
    
    for(int i=0; i<arr2.size(); i++){
        string tmp;
        while(arr2[i]){
            tmp += (arr2[i]%2+'0');
            arr2[i] /=2;
        }
        reverse(tmp.begin(),tmp.end());
        while(tmp.size() != n){
            tmp = '0'+tmp;
        }
        // cout << tmp << '\n';
        
        for(int j=0; j<n; j++){
            if(tmp[j] == '1'){
                answer[i][j] = '#';
            }
        }
    }
    return answer;
}
```

#include <bits/stdc++.h>


using namespace std;

vector<string> solution(int n, vector<int> arr1, vector<int> arr2) {
    vector<string> answer;
    for(int i=0; i<n; i++){
        arr1[i] = arr1[i] | arr2[i];
        string tmp;
        for(int j=0; j<n; j++){
            if(arr1[i] % 2){
                tmp = '#'+tmp;
            }else {
                tmp = ' '+tmp;
            }
            arr1[i] /=2;
        }
        answer.push_back(tmp);
    }
    return answer;
}
