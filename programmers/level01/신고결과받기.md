# 프로그래머스 Lv.1 신고 결과 받기

## Solution 1

```c++
#include <bits/stdc++.h>


using namespace std;

vector<int> solution(vector<string> id_list, vector<string> report, int k) {
    vector<int> answer;
    unordered_map<string , int> report_count;
    unordered_map<string , set<string> > report_list;
    unordered_map<string , set<string> > mp;
    
    for(auto &re : report) {
        stringstream iss(re);
        string a ,b;
        iss >> a >> b;
        report_list[a].insert(b);
        auto s = mp[b];
        if(s.find(a) == s.end()){
            report_count[b]++;
        }
        mp[b].insert(a);
    }
    
    for(auto &id : id_list) {
        auto list = report_list[id];
        int count = 0;
        for(auto l : list){
            if(report_count[l] >= k) count++;
        }
        answer.push_back(count);
    }
    return answer;
}
```


## Solution 2

```c++
#include <bits/stdc++.h>


using namespace std;

vector<int> solution(vector<string> id_list, vector<string> report, int k) {
    int n = id_list.size();
    vector<int> answer(n);
    vector<int> cnt(n);
    unordered_map<string , int> mp;
    vector<pair<int ,int>> v;
    
    for(int i=0; i< n; i++) {
        mp[id_list[i]] = i;
    }
   
    
    for(auto &re : report) {
        stringstream iss(re);
        string a ,b;
        iss >> a >> b;
        v.push_back({mp[a], mp[b]});
    }
    sort(v.begin() , v.end());
    v.erase(unique(v.begin() , v.end()) , v.end());
    
    int sz = v.size();
    for(int i=0; i<sz; i++) {
        cout << v[i].first << ' ' << v[i].second << '\n';
        int r = v[i].second;
        cnt[r]++;
    }
    for(const auto& [a,b] : v){
        if(cnt[b] >= k) answer[a]++;
    }
    return answer;
}
```

```c++
using namespace std;

vector<int> solution(vector<string> id_list, vector<string> report, int k) {
    vector<int> answer;
    unordered_map<string, unordered_set<string>> report_list;
    unordered_map<string ,int> reports;
    string a,b;
    for(string r : report){
        stringstream iss(r);
        iss >> a >> b;
        if(report_list[a].find(b) == report_list[a].end()){
            report_list[a].insert(b);
            reports[b]++;
        }
       
    }
    for(auto id : id_list){
        auto s = report_list[id];
        int count = 0;
        for(auto i : s){
            if(reports[i] >= k) count++;
        }
        answer.push_back(count);
    }
    return answer;
}
```

```c++
#include <bits/stdc++.h>

using namespace std;

vector<int> solution(vector<string> id_list, vector<string> report, int k) {
    vector<int> answer;
    sort(report.begin(), report.end());
    report.erase(unique(report.begin(),report.end()), report.end());
    unordered_map<string, vector<string>> report_list;
    unordered_map<string ,int> reports;
    string a,b;
    for(string r : report){
        stringstream iss(r);
        iss >> a >> b;
        reports[b]++;
        report_list[a].push_back(b);   
    }
    for(auto id : id_list){
        auto s = report_list[id];
        int count = 0;
        for(auto i : s){
            if(reports[i] >= k) count++;
        }
        answer.push_back(count);
    }
    return answer;
}
```
