# 프로그래머스 Lv.1 성격유형

## solution 1

```c++
#include <bits/stdc++.h>

using namespace std;

string solution(vector<string> survey, vector<int> choices) {
    string answer = "";
    string mbt1 = "RCJA";
    string mbt2 ="TFMN";
    unordered_map<char,int > mp;
    int len = survey.size();
    for(int i=0; i<len; i++) {
        if(choices[i] >= 4){
            mp[survey[i][1]] += choices[i]-4;
        }else {
            mp[survey[i][0]] += 4-choices[i];
        }
    }
    for(int i=0; i<4; i++){
        if(mp[mbt2[i]] > mp[mbt1[i]]){
            answer += mbt2[i];
        }else {
            answer += mbt1[i];
        }
    }
    return answer;
}
```
