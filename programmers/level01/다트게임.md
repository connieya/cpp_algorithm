# 프로그래머스 Lv.1 다트게임

## solution 1

```c
#include <bits/stdc++.h>

using namespace std;


int solution(string dartResult) {
    int answer = 0;
    vector<int> v;
    int res = 0;
    for(char ch : dartResult){
        if(isdigit(ch)){
            res = res*10 + (ch-'0');
        }else if(ch == 'S'){
            v.push_back(res);
            res = 0;
        }else if(ch == 'D'){
            v.push_back(res*res);
            res = 0;
        }else if(ch == 'T'){
            v.push_back(res*res*res);
            res = 0;
        }else if(ch == '*'){
            if(v.size() > 1){
                v.back() *=2;
                v[v.size()-2] *= 2;
            }else {
                v.back() *= 2;
            }
        }else {
            v.back() = -v.back();
        }
    }
    for(auto a : v) answer += a;
    return answer;
}
```
