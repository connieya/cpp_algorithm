# 프로그래머스 Lv.1 크레인 인형뽑기 게임

## solution 1

```c++
#include "bits/stdc++.h"

using namespace std;

int solution(vector<vector<int>> board, vector<int> moves) {
    int answer = 0;
    int len = board.size();
    stack<int> bucket;
    for(int move : moves) {
        for(int i=0; i<len; i++){
            cout << move << ' ' << board[i][move-1] << '\n';
            if(board[i][move-1] > 0) {
                if(!bucket.empty() &&  bucket.top() == board[i][move-1]){
                    bucket.pop();
                    answer += 2;
                }else {
                    bucket.push(board[i][move-1]);
                }
                board[i][move-1] = 0;
                break;
            }
        }
    }
    return answer;
}
```

## solution 2

```c++
#include <bits/stdc++.h>



using namespace std;


int solve(vector<vector<int>> &board , int dir , int n) {
    int start = 0;
    int res  = 0;
    while(start < n){
        if(board[start][dir]){
            res = board[start][dir];
            board[start][dir] = 0;
            break;
        }
        start++;
    }
    return res;
}

int solution(vector<vector<int>> board, vector<int> moves) {
    int answer = 0;
    stack<int> s;
    for(int move  : moves){
        int res = solve(board,move-1, board.size());
        if(!res) continue;
        if(!s.empty() && s.top() == res){
            answer += 2;
            s.pop();
            continue;
        }
        s.push(res);
    }
    return answer;
}
```
