# 프로그래머스 Lv.4 지형 이동

## solution 1

```c
#include <bits/stdc++.h>


using namespace std;

bool check[301][301];
int arr[301][301];
int dx[] = {-1,0,1,0};
int dy[] = {0,1,0,-1};
int n;
int parent[100001];
vector<vector<int >> L;
vector<tuple<int,int,int>> v;

void bfs(int x ,int y , int num , int h ) {
    queue<pair<int,int>> q;
    q.push({x,y});
    arr[x][y] = num;
    while(!q.empty()) {
        auto cur = q.front();
        x = cur.first;
        y = cur.second;
        q.pop();
        for(int i=0; i<4; i++){
            int nx = x +dx[i];
            int ny = y + dy[i];
            if(nx < 0 || nx >= L.size() || ny < 0 || ny >= L.size() ) continue;
            if(!arr[nx][ny] && abs(L[x][y]-L[nx][ny]) <= h ){
                q.push({nx,ny});
                arr[nx][ny] = num;
            }
        }
    }
}

void make_connect(int x, int y){
    for(int i=0; i<4; i++){
        int nx = x +dx[i];
        int ny = y +dy[i];
        if(nx < 0 || nx >= L.size() || ny < 0 || ny >= L.size()) continue;
        if(arr[x][y] != arr[nx][ny]){
            v.push_back({ abs(L[x][y]-L[nx][ny]) , arr[x][y] , arr[nx][ny] });
        }
    }
    
    }


int find(int a){
    if(a == parent[a]) return a;
    return parent[a] = find(parent[a]);
}

void Union(int a, int b){
    a= find(a);
    b= find(b);
    if(a != b){
        parent[a] = b;
    }
}


int solution(vector<vector<int>> land, int height) {
    int answer = 0;
    int num = 1 , n = land.size();
    L = land;
    for(int i=0; i < n; i++){
        for(int j=0; j < n; j++) {
            if(!arr[i][j]){
                bfs(i,j,num , height);
                num++;
            }
        }
    }
    
    for(int i=0; i<num; i++){
        parent[i] = i;
    }
    

    for(int i=0; i<n; i++){
        for(int j=0; j<n; j++){
           make_connect(i,j); 
        }
    }
    
    sort(v.begin() , v.end());
    int x,y,cost,cnt = 0;
    for(int i=0; i<v.size(); i++){
        tie(cost,x,y) = v[i];
        if(find(x) != find(y)){
            Union(x,y);
            answer += cost;
            cnt++;
        }
        if(cnt == num -1) break;
    }
    
    return answer;
}
```
