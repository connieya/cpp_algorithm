# 프로그래머스 Lv.4

## solution 1

```c
#include <bits/stdc++.h>

using namespace std;

int countRocks(int dis , vector<int> &rocks) {
    int cnt = 0;
    int prev = 0;
    int len = rocks.size();
    for(int i=0; i < len; i++) {
        if(rocks[i] - prev < dis){
              cnt++;
        }else {
            prev = rocks[i];
        }
    }
    return cnt;
}

int solution(int distance, vector<int> rocks, int n) {
    int answer = INT_MAX;
    int len = rocks.size();
    sort(rocks.begin() , rocks.end());
    rocks.push_back(distance);
    int lt = 0 , rt = distance;
    while(lt <= rt) {
        int mid = (lt+rt)/2;
        int cnt = countRocks(mid , rocks);
        if(cnt <= n){
            lt = mid+1;
            answer = mid;
        }else {
            rt = mid-1;
        }
    }
    
    return answer;
}
```
