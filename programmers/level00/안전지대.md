# 프로그래머스 Lv.0 안전 지대

## solution 1

```c++
#include <bits/stdc++.h>

using namespace std;

int len;
int dx[] = {-1,-1,-1,0,1,1,1,0};
int dy[] = {-1,0,1,1,1,0,-1,-1};
bool visited[101][101];

int solution(vector<vector<int>> board) {
    int answer = 0;
    len = board.size();
    for(int i=0; i<len; i++){
        for(int j=0; j<len; j++){
            if(board[i][j] == 1){
                visited[i][j] = true;
                for(int k=0; k<8; k++){
                    int nx = dx[k]+i;
                    int ny = dy[k]+j;
                    if(nx < 0 || nx >= len || ny < 0 || ny >=len) continue;
                    visited[nx][ny] = true;
                }
            }
        }
    }
    
    for(int i=0; i<len; i++){
        for(int j=0; j<len; j++){
            answer += !visited[i][j];
        }
    }
    return answer;
}
```
