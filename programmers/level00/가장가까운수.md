# 프로그래머스 Lv.0 가까운 수

## solution 1 sort

```c++
#include "bits/stdc++.h"
using namespace std;

int solution(vector<int> array, int n) {
    sort(array.begin() , array.end());
    int mn = 1000;
    int answer =0;
    for(auto a : array){
      if(abs(n-a) < mn){
          mn = abs(n-a);
          answer = a;
      }  
    }
   
    return answer;
}
```
