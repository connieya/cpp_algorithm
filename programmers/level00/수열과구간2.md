# 프로그래머스 Lv.0 수열과 구간 쿼리 2


## solution 1  for문

```c++
#include "bits/stdc++.h"

using namespace std;

vector<int> solution(vector<int> arr, vector<vector<int>> queries) {
    vector<int> answer;
    for(auto query : queries) {
        int s = query[0];
        int e = query[1];
        int k = query[2];
        int ans = 1e6+1;
        for(int i=s; i<=e; i++){
            if(arr[i] <=k) continue;
            ans = min(ans , arr[i]);
        }
        if(ans == 1e6+1){
            answer.push_back(-1);
        }else {
            answer.push_back(ans);
        }
    }
    return answer;
}
```
