# 프로그래머스 Lv.0 이진수 더하기 

## solution 1 구현 

```c++
#include <string>
#include <vector>

using namespace std;

string solution(string bin1, string bin2) {
    string answer = "";
    int n = bin1.length()-1;
    int m = bin2.length()-1;
    bool add = false;
    while (n >=0 && m >=0){
        if(bin1[n] == '1' && bin2[m] == '1'){
            if(add) {
                answer = '1' + answer;
            }else {
                answer = '0' +answer;    
            }
            
            add = true;
        }else if(bin1[n] == '0' && bin2[m] == '0') {
            if(add) {
                answer = '1' + answer;
            }else {
                answer = '0' + answer ;
            }
            add = false;
            
        }else {
            if(add){
                answer = '0' + answer ;
                add =true;
            }else {
                answer = '1' + answer;
                add = false;
            }
        }
        n--;
        m--;
    }
    while (n >=0){
        if(bin1[n] == '0'){
            if(add){
                answer = '1' + answer ;
                add = false;
            }else {
                answer =  '0' + answer;
                add = false;
            }
        }else {
            if(add){
                answer = '0' + answer;
                add = true;
            }else {
                answer = '1'  + answer;
                add = false;
            }
        }
        n--;
    }
    while (m >=0){
        if(bin2[m] == '0'){
            if(add){
                answer = '1' + answer ;
                add = false;
            }else {
                answer =  '0' + answer;
                add = false;
            }
        }else {
            if(add){
                answer = '0' + answer;
                add = true;
            }else {
                answer = '1'  + answer;
                add = false;
            }
        }
        m--;
    }
    if(add){
        answer = '1' + answer;
    }
    return answer;
}
```
