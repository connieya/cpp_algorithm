# 프로그래머스 Lv.0 1로 만들기

## solution 1 구현

```c++
#include "bits/stdc++.h"

using namespace std;

int solution(vector<int> num_list) {
    int answer = 0;
    for(auto num : num_list) {
        int cnt = 0;
        while(num > 1) {
            if(num % 2) {
                num = (num-1)/2;
            }else {
                num /=2;
            }
            cnt++;
        }
        answer += cnt;
    }
    return answer;
}
```
