# 프로그래머스 Lv.3 보석 쇼핑

## Solution1 

```c
#include <bits/stdc++.h>

using namespace std;

vector<int> solution(vector<string> gems) {
    unordered_map<string,bool> mp;
    unordered_map<string,int> mp2;
    int total = 0;
    for(string gem :gems){
        if(mp[gem] == false){
            total++;
            mp[gem] = true;
        }
    }
    int lt = 0 , rt = 0 , cnt = 0 , mn = INT_MAX;
    vector<int> answer;
    while(rt < gems.size()){
        if(cnt < total){
            if(mp2[gems[rt]]== 0) cnt++;
            mp2[gems[rt]]++;
            rt++;
        }else {
            int sz = rt-lt;
            if(mn > sz){
                mn = sz;
                answer.clear();
                answer.push_back(lt+1);
                answer.push_back(rt);
            }
            mp2[gems[lt]]--;
            if(mp2[gems[lt]] == 0) cnt--;
            lt++;
        }
    }
    if(cnt == total){
        while(mp2[gems[lt]] > 1){
            mp2[gems[lt]]--;
            lt++;
        }
        if(mn > rt-lt){
            answer.clear();
            answer.push_back(lt+1);
            answer.push_back(rt);
        }
    }
 
    return answer;
}
```
