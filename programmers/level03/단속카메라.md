# 프로그래머스 Lv.3 단속카메라

## solution 1

```c++
#include <bits/stdc++.h>
using namespace std;

bool compare(vector<int> a , vector<int> b ){
   return a[1] < b[1];
}

int solution(vector<vector<int>> routes) {
    int answer = 1;
    sort(routes.begin(),routes.end() ,compare);
    int len = routes.size();
    int prev = routes[0][1];
    for(int i=1; i<routes.size(); i++){
        int now = routes[i][0];
        if(prev < now){
            answer++;
            prev = routes[i][1];
        } 
    }
    return answer;
}
```
