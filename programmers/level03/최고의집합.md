# 프로그래머스  Lv.3 최고의 집합

## solution 1

```c
#include <bits/stdc++.h>

using namespace std;

vector<int> solution(int n, int s) {
    vector<int> answer;
    int a = s % n;
    int b = s / n;
    if(!b) return {-1};
    for(int i=0; i< (n-a); i++){
        answer.push_back(b);
    }
    for(int i=0; i<a; i++){
        answer.push_back(b+1);
    }
    sort(answer.begin(),answer.end());
    return answer;
  
}
```

## solution 2

```c
#include <bits/stdc++.h>

using namespace std;

vector<int> solution(int n, int s) {
    vector<int> answer;
    if(!(s/n)) return {-1};
    while(n){
        answer.push_back(s/n);
        s -= s/n;
        n--;
    }
    sort(answer.begin(),answer.end());
    return answer;
}
```
