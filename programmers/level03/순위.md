# 프로그래머스 Lv.3 순위

## Solution 1

```c
#include <bits/stdc++.h>

using namespace std;

bool arr[101][101];

int solution(int n, vector<vector<int>> results) {
    int answer = 0;
    for(auto res : results) {
        arr[res[0]][res[1]] =1;
    }
    for(int k=1; k<=n; k++) {
        for(int i=1; i<=n; i++) {
            for(int j=1; j<=n; j++) {
                if(arr[i][k] && arr[k][j]) arr[i][j] = 1;
            }
        }
    }
    for(int i=1; i<=n; i++){
        int cnt = 0;
        for(int j=1; j<=n; j++){
            cnt +=  (arr[i][j] || arr[j][i] );
        }
        if(cnt == n-1) answer++;
    }
    return answer;
}
```
