# 프로그래머스 Lv.3 순위

## Solution 1

```c
#include <bits/stdc++.h>

using namespace std;

bool arr[101][101];

int solution(int n, vector<vector<int>> results) {
    int answer = 0;
    for(auto res : results) {
        arr[res[0]][res[1]] =1;
    }
    for(int k=1; k<=n; k++) {
        for(int i=1; i<=n; i++) {
            for(int j=1; j<=n; j++) {
                if(arr[i][k] && arr[k][j]) arr[i][j] = 1;
            }
        }
    }
    for(int i=1; i<=n; i++){
        int cnt = 0;
        for(int j=1; j<=n; j++){
            cnt +=  (arr[i][j] || arr[j][i] );
        }
        if(cnt == n-1) answer++;
    }
    return answer;
}
```

## solution 2

```c
#include <bits/stdc++.h>

using namespace std;

vector<int> win[101];
vector<int> lose[101];
bool win_visited[101];
bool lose_visited[101];

void w_dfs(int start ,int prev){
    for(int next : win[start]){
        if(win_visited[next]) continue;
        win_visited[next] = true;
        w_dfs(next,start);
    }

}

void l_dfs(int start ,int prev){
    for(int next : lose[start]){
        if(lose_visited[next]) continue;
        lose_visited[next] = true;
        l_dfs(next,start);
    }

}

void init(int n){
    for(int i=1; i<=n; i++){
        win_visited[i] = false;
        lose_visited[i] = false;
    }
}

int solution(int n, vector<vector<int>> results) {
    int answer = 0;
    for(auto &result : results){
        win[result[0]].push_back(result[1]);
        lose[result[1]].push_back(result[0]);
    }
    for(int i=1; i<=n; i++){
        w_dfs(i,-1);
        l_dfs(i,-1);
        int cnt = 0;
        for(int i=1; i<=n; i++){
            cnt += win_visited[i];
            cnt += lose_visited[i];
        }
        init(n);
        answer += (cnt == n-1);
    }
    return answer;
}
```
