# 프로그래머스 Lv.3 네트워크

## solution 1

```c
#include <bits/stdc++.h>

using namespace std;

bool visited[201];
vector<int> graph[201];

void dfs(int start) {
    visited[start] = true;
    for(int next : graph[start]){
        if(!visited[next]){
            dfs(next);
        }
    }
}

int solution(int n, vector<vector<int>> computers) {
    int answer = 0;
    int len = computers[0].size();
    for(int i=0; i<computers.size(); i++) {
        for(int j=0; j <len; j++) {
            if(computers[i][j]){
                graph[i].push_back(j);
            }
        }
    }
    
    for(int i=0; i<n; i++){
        if(!visited[i]){
            dfs(i);
            answer++;
        }
    }
    return answer;
}
```
