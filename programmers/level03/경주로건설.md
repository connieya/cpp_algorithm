# 프로그래머스 Lv.3 경주로 건설

## solution 1

```c
#include <bits/stdc++.h>

using namespace std;

struct Car {
    int x,y,cost,head;
};

int dx[] = {-1,0,1,0};
int dy[] = {0,1,0,-1};
bool check[26][26];
int cost[26][26];
vector<vector<int>> mp;
int n,ans = 1e8;


void bfs(Car car){
    queue<Car> q;
    check[car.x][car.y] = true;
    q.push({0,0,0,5});
    while(!q.empty()) {
        auto cur = q.front();
        q.pop();
        if(cur.x == n-1 && cur.y == n-1 && cur.cost <= ans){
            ans = cur.cost;
            continue;
        }
        for(int i=0; i<4; i++){
            int nx = dx[i] + cur.x;
            int ny = dy[i]+ cur.y;
            if(nx < 0 || nx >=n || ny < 0 || ny >= n || mp[nx][ny]) continue;
            int cur_cost = cur.cost;
            if(i == cur.head || cur.head == 5)cur_cost += 100;
            else cur_cost += 600;
            
            if(!check[nx][ny] || cost[nx][ny] >= cur_cost-400) {
                check[nx][ny] = 1;
                cost[nx][ny] = cur_cost;
                q.push({nx,ny,cur_cost ,i});
            }
            
        }
    }
    
}

int solution(vector<vector<int>> board) {
    int answer = 0;
    n = board.size();
    mp = board;
    bfs({0,0,0,5});
    return ans;
}
```
