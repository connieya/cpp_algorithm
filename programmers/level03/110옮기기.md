# 프로그래머스 Lv.3 110 옮기기

## solution 1

```c++
#include <bits/stdc++.h>

using namespace std;

vector<string> solution(vector<string> strs) {
    vector<string> answer;
    for(auto str : strs){
        stack<char> ch;
        int cnt = 0;
        for(auto s : str){
            ch.push(s);
            if(ch.size() >= 3){
                char third = ch.top();
                ch.pop();
                char second = ch.top ();
                ch.pop();
                char first = ch.top();
                ch.pop();
                if(first == '1' && second == '1' && third == '0'){
                    cnt++;
                }else {
                    ch.push(first);
                    ch.push(second);
                    ch.push(third);
                }
            }
        }
        
        if(!cnt){
            answer.push_back(str);
            continue;
        }
        int idx = ch.size();
        string temp = "";
        while(!ch.empty() && ch.top() == '1'){
            --idx;
            temp = ch.top() + temp;
            ch.pop();
        }
        while(!ch.empty()){
            temp = ch.top()+temp;
            ch.pop();
        }
        while(cnt-- > 0){
            temp.insert(idx,"110");
        }
        answer.push_back(temp);
    }
    return answer;
}
```
