# 프로그래머스 Lv.3  불량 사용자

## Solution 1

```c
#include <bits/stdc++.h>

using namespace std;

vector<string> user , banned;
bool checked[9];
bool visited[260];
int len , user_size;

int get_bit_num(){
    int ret = 0;
    for(int i=0; i<user_size; i++){
        if(checked[i]) ret += 1<<i;
    }
    return ret;
}

int dfs(int L){
    if(L == len) {
        int num = get_bit_num();
        if(!visited[num]){
            visited[num] = true;
            return 1;
        }
        return 0;
    }
    int res = 0;
    for(int i=0; i<user_size; i++) {
        string user_id = user[i];
        string banned_id = banned[L];
        if(user_id.size() != banned_id.size()) continue;
        bool flag = true;
        for(int j=0; j < user_id.size(); j++) {
            if(banned_id[j] == '*') continue;
            if(banned_id[j] != user_id[j]){
                flag = false;
                break;
            }
        }
        if(flag && !checked[i]){
            checked[i] = true;
            res += dfs(L+1);
            checked[i] = false;
        }
    }
    return res;
}

int solution(vector<string> user_id, vector<string> banned_id) {
    user = user_id;
    banned = banned_id;
    len = banned_id.size();
    user_size = user_id.size();
    return dfs(0);
}
```
