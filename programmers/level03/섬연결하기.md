# 프로그래머스 Lv.3 섬 연결하기

## solution 1

```c++
#include <bits/stdc++.h>

using namespace std;

int parent[101];

int find(int a){
    if(a == parent[a]) return a;
    return parent[a] = find(parent[a]);
}

void Union(int a ,int b){
    a = find(a);
    b = find(b);
    if(a !=  b) {
        parent[a] = b;
    }
}


int solution(int n, vector<vector<int>> costs) {
    int answer = 0;
    int cnt = 0;
    vector<tuple<int,int,int>> graph;
    for(int i=0; i<n; i++){
        parent[i] = i;
    }
    for(auto cost : costs){
        graph.push_back({cost[2],cost[0],cost[1]});
    }
    
    sort(graph.begin(), graph.end());
    for(auto g  : graph){
        int cost , a, b;
        tie(cost,a,b) = g;
        if(find(a) != find(b)){
            answer += cost;
            Union(a,b);
            cnt++;
        }
        if(cnt ==n-1)break;
    }
    return answer;
}
```

## solution 2

```c++
#include <bits/stdc++.h>

using namespace std;

int parent[101];

int find(int a){
    if(a == parent[a]) return a;
    return parent[a] = find(parent[a]);
}

void Union(int a ,int b){
    a = find(a);
    b = find(b);
    if(a !=  b) {
        parent[a] = b;
    }
}

bool compare(vector<int> &a , vector<int> &b){
    return a[2] < b[2];
}


int solution(int n, vector<vector<int>> costs) {
    int answer = 0;
    int cnt = 0;
    vector<tuple<int,int,int>> graph;
    for(int i=0; i<n; i++){
        parent[i] = i;
    }
    sort(costs.begin(), costs.end(), compare);
    for(auto cost  : costs){
        int a = cost[0];
        int b = cost[1];
        int c = cost[2];
        if(find(a) != find(b)){
            answer += c;
            Union(a,b);
            cnt++;
        }
        if(cnt ==n-1)break;
    }
    return answer;
}
```
