```c
        res.pop_back();

        string tmp;
        tmp += str;
        tmp += next;
        if(!visited[tmp]){
            res.push_back(str);
            visited[tmp] = true;
            dfs(mp[next] , next, L+1);
            visited[tmp] = false;
            res.pop_back();
        }
    }
}

        visited[tmp] = false;

    int idx = 0;

    for(auto ticket : tickets){
        graph[mp[ticket[0]]].push_back(ticket[1]);
    }
    
    dfs(mp["ICN"],"ICN",0);
```

## solution 2

### 너무 어렵게 생각함... 문자열에 너무 신경을 많이 씀,,, 그냥 인덱스로 해서 dfs 하면 끝

```c
#include <bits/stdc++.h>


using namespace std;

bool visited[10001];
vector<vector<string>> answer;
vector<string> res;
int n;

void dfs(string start , vector<vector<string>> &tickets , int L) {
    if(L == n){
        res.push_back(start);
        answer.push_back(res);
        res.pop_back();
        return;
    }
    
    for(int i=0; i<tickets.size(); i++){
        if(!visited[i] && tickets[i][0] == start){
            res.push_back(start);
            visited[i] = true;
            dfs(tickets[i][1], tickets, L+1);
            visited[i] = false;
            res.pop_back();
        } 
    }
    
}

vector<string> solution(vector<vector<string>> tickets) {
    n = tickets.size();
    dfs("ICN", tickets, 0);
    sort(answer.begin(), answer.end());
    return answer[0];
}
```

```c
#include <bits/stdc++.h>


using namespace std;

bool visited[10001];
string res = "a";
int n;

void dfs(string start , string cur , vector<vector<string>> &tickets , int L) {
    if(L == n){
        if(res >  cur){
            res = cur;
        }
        return;
    }
    
    for(int i=0; i<tickets.size(); i++){
        if(!visited[i] && tickets[i][0] == start){
            visited[i] = true;
            dfs(tickets[i][1], cur+tickets[i][1] , tickets, L+1);
            visited[i] = false;
        } 
    }
    
}

vector<string> solution(vector<vector<string>> tickets) {
    n = tickets.size();
    dfs("ICN","ICN" , tickets, 0);
    vector<string> a;
    for(int i=0; i < res.size(); i+=3){
        a.push_back(res.substr(i,3));
    }
    return a;
}
```
