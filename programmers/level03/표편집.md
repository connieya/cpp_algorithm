# 프로그래머스 Lv.3  표 편집

## Solution 1

```c++
#include <bits/stdc++.h>
using namespace std;
list<int> l;
list<int> ::iterator its[10000015];
list<int> ::iterator cursor;
stack<pair<int,int>> erased;

string solution(int n, int k, vector<string> cmd) {
    for(int i=0; i<= n; i++){
        l.push_back(i);
    }
    auto it = l.begin();
    for(int i= 0; i <=n; i++){
        its[i] = it;
        it++;
    }
    
    cursor = its[k];
    for(auto c : cmd) {
        if(c[0] == 'D') {
            int num = stoi(c.substr(2,100));
            while(num--){
                cursor++;
            }
        }else if(c[0] == 'C') {
            erased.push({*cursor , *(next(cursor))});
            cursor = l.erase(cursor);
            if(*cursor == n ) cursor--;
        }else if(c[0] == 'U') {
            int num = stoi(c.substr(2,100));
            while(num--){
               cursor--;
            }
        }else {
            int cur , nxt;
            tie(cur,nxt) = erased.top();
            erased.pop();
            its[cur] = l.insert(its[nxt] , cur);
        }
    }
    string answer(n,'X');
    for(auto x : l ){
        if(x != n) answer[x] = 'O';
    }
    return answer;
}
```

## Solution 2

```c++
#include <bits/stdc++.h>
using namespace std;

const int MX = 1000005;
int pre[MX] , nxt[MX];
 
string solution(int n, int k, vector<string> cmd) {
    string answer(n,'O');
    for(int i=0; i< n; i++) {
        pre[i] = i-1;
        nxt[i] = i+1;
    }
    nxt[n-1] = -1;
    stack<tuple<int,int,int>> erased;
    int cursor = k;
    for(auto &c : cmd) {
        if(c[0] == 'U'){
            int num = stoi(c.substr(2,10));
            while(num--){
                cursor = pre[cursor];
            }
        }else if(c[0] == 'D') {
            int num = stoi(c.substr(2,10));
            while(num--) {
                cursor = nxt[cursor];
            }
        }else if(c[0] == 'C') {
            erased.push({pre[cursor] , cursor , nxt[cursor]});
            if(pre[cursor] != -1) {
                nxt[pre[cursor]] = nxt[cursor];
            }
            if(nxt[cursor] != -1) {
                pre[nxt[cursor]] = pre[cursor];
            }
            answer[cursor] = 'X';
            if(nxt[cursor] != -1){
                cursor = nxt[cursor];
            }else {
                cursor = pre[cursor];
            }
        }else {
            int prev , cur , next;
            tie(prev ,cur , next) = erased.top();
            erased.pop();
            if(prev != -1) {
                nxt[prev] = cur;
            }
            if(next != -1) {
                pre[next] = cur;
            }
            answer[cur] = 'O';
        }
    }
    return answer;
}
```

## Solution 3

```c++
#include <bits/stdc++.h>
using namespace std;

typedef struct Node{
    int data;
    Node * next;
    Node * prev;
};
 
string solution(int n, int k, vector<string> cmd) {
    string answer =  "";
    stack<int> deleted;
    vector<Node*> node(n);
    
    for(int i=0; i< n; i++){
        node[i] = new Node();
        node[i]->data = i;
    }
    for(int i=0; i<n; i++){
        if(i> 0){
            node[i]->prev = node[i-1];
        }
        if(i < n-1){
            node[i]->next = node[i+1];
        }
    }
    
    Node* curr = node[k];
    
    vector<bool> remained(n , true);
    for(auto &c : cmd) {
        if(c[0] == 'U') {
            int num = stoi(c.substr(2));
            while(num--){
                curr = curr->prev;
            }
        }else if(c[0] == 'D'){
            int num = stoi(c.substr(2));
            while(num--){
                curr = curr->next;
            }
        }else if(c[0] == 'C') {
            deleted.push(curr->data);
            remained[curr->data] = false;
            
            if(curr->prev != NULL){
                curr->prev->next = curr->next;
            }
            if(curr->next != NULL){
                curr->next->prev = curr->prev;
            }
            
            if(curr->next != NULL) {
                curr = curr->next;
            }else {
                curr =curr->prev;
            }
        }else {
            int row = deleted.top();
            deleted.pop();
            remained[row] = true;
            if(node[row]->prev != NULL) {
                node[row]->prev->next = node[row];
            }
            if(node[row]->next != NULL) {
                node[row]->next->prev = node[row];
            }
        }
    }
    for(int i=0; i<n; i++){
        answer += remained[i] ? "O" : "X";
    }
    
    return answer;
}
```
