# Leetcode 316. Remove Duplicate Letters


## Solution 1

```c++
class Solution {
public:
    string removeDuplicateLetters(string s) {
        vector<int> alpha(26,0);
        vector<int> visited(26,false);
        
        for(char ch : s){
            alpha[ch-'a']++;
        }
        string result = "";
        for(char ch : s) {
            alpha[ch-'a']--;
            if(visited[ch-'a']) continue;
            
            while(ch < result.back() && alpha[result.back()-'a']) {
                visited[result.back()-'a'] = false;
                result.pop_back();
            }
            result += ch;
            visited[ch-'a'] = true;
        }
        
        return result;
    }
};
```


## Solution 2

```c++
class Solution {
public:
    string removeDuplicateLetters(string s) {
        if(s.empty()) return s;
        int cnt[26] = {};
        for(char ch : s){
            cnt[ch-'a']++;
        }
        int best = 0;
        for(int i=0; i<s.size(); i++) {
            if(s[i] < s[best]) best = i;
            if(--cnt[s[i]-'a'] == 0) break;
        }
        string t = s.substr(best +1);
        t.erase(remove(t.begin() , t.end() , s[best]) , t.end());
        return s[best] + removeDuplicateLetters(t);
    }
};
```

## Solution 3


```c++
class Solution {
public:
    string removeDuplicateLetters(string s) {
        string ans;
        int last[26] = {};
        vector<bool> checked(26,false);
        int len = s.size();
        for(int i =0; i< len; i++){
            last[s[i]-'a'] = i;
        }
        
        for(int i=0; i< len; i++){
            if(checked[s[i]-'a']) continue;
            while(ans.size() && ans.back() > s[i]  && i < last[ans.back() -'a']) {
                checked[ans.back()-'a'] = false;
                ans.pop_back();
            }
            ans.push_back(s[i]);
            checked[s[i]-'a'] = true;
        }
        
        return ans;
    }
};
```
