# LeetCode 46. Permutations

## Solution 1

```c
class Solution {
    bool visited[7];
    int arr[7];
    vector<int> ans;
    vector<vector<int>> res;
     
    void dfs(int L , int len) {
        if(L == len) {
            vector<int> tmp;
            for(int i=0; i<len; i++){
                tmp.push_back(ans[arr[i]]);
            }
            res.push_back(tmp);
            return;
        }
        for(int i=0; i< len; i++) {
            if(!visited[i]){
                visited[i] = true;
                arr[L] =  i;
                dfs(L+1 ,len);
                visited[i] = false;
            }
        }
    }
    
public:
    vector<vector<int>> permute(vector<int>& nums) {
        int len = nums.size();
        ans = nums;
        dfs(0,len); 
        return res;
    }
};
```

```c
class Solution {
    
public:
    vector<vector<int>> permute(vector<int>& nums) {
        vector<vector<int>> result;
        dfs(nums,0,result); 
        return result;
    }
    
    void dfs(vector<int> &num , int begin , vector<vector<int >> &result) {
        if(begin >= num.size()) {
            result.push_back(num);
            return;
        }
        
        for(int i= begin; i < num.size(); i++) {
            swap(num[begin] , num[i]);
            dfs(num , begin+1 , result);
            swap(num[begin], num[i]);
        }
    }
};
```

## Solution 4

```c
class Solution {
    
public:
    vector<vector<int>> permute(vector<int>& nums) {
        vector<vector<int>> result;
        sort(nums.begin(), nums.end());
        do {
            result.push_back(nums);            
        }while(next_perm(nums));
        return result;
    }
    
    bool next_perm(vector<int> &nums) {
        int i = nums.size() -1;
        while(i > 0 && nums[i-1] > nums[i]){
            i--;
        }
        if(!i) return false;
        
        int j = nums.size()-1;
        while(nums[i-1] > nums[j]) {
            j--;
        }
        
        int temp = nums[j];
        nums[j] = nums[i-1];
        nums[i-1] = temp;
        
        j = nums.size()-1;
        while(i < j) {
            temp = nums[j];
            nums[j] = nums[i];
            nums[i] = temp;
            i++;
            j--;
        }
        return true;   
    }
};
```
