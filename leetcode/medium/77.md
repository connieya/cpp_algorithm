# Leetcode 77. Combinations

## solution 1

```c
class Solution {
    vector<int> tmp;
    vector<vector<int>> ans;
    void comb(int n ,int L ,int k, int start){
        if(L == k){
            ans.push_back(tmp);
            return;
        }
        
        for(int i = start; i <=n; i++){
            tmp.push_back(i);
            comb(n,L+1,k,i+1);
            tmp.pop_back();
        }
    }
public:
    vector<vector<int>> combine(int n, int k) {
        comb(n,0,k,1);
        return ans;
    }
};
```
