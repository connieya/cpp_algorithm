# 819. Most Common Word

## Solution 1

### Runtime : 36 ms , Memory : 7.6 MB

```c++
class Solution {
public:
    string mostCommonWord(string paragraph, vector<string>& banned) {
        transform(paragraph.begin() , paragraph.end() , paragraph.begin() , ::tolower);
        unordered_map<string,int> mp;
        string tmp = "";
        for(int i =0; i < paragraph.size(); i++) {
            if(isalpha(paragraph[i])){
                tmp.push_back(paragraph[i]);
            }else {
                if(tmp.length() > 0){
                    mp[tmp]++;
                }
                tmp.clear();
            }
        }
        if(tmp.length() > 0) mp[tmp]++;
        for(string &b : banned){
            mp[b] = 0;
        }
        int cnt = 0;
        string res = "";
        for(auto m : mp){
            if(m.second > cnt){
                cnt = m.second;
                res = m.first;
            }
        }
        return res;
    }
};
```

class Solution {
public:
    string mostCommonWord(string paragraph, vector<string>& banned) {
        unordered_map<string,int> mp;
        for(int i =0; i < paragraph.size();) {
            string tmp = "";
            while(i < paragraph.size() && isalpha(paragraph[i])){
                tmp.push_back(tolower(paragraph[i++]));
            }
            while(i< paragraph.size() && !isalpha(paragraph[i])){
                i++;
            }
            mp[tmp]++;
        }
        for(string &b : banned){
            mp[b] = 0;
        }
        int cnt = 0;
        string res = "";
        for(auto m : mp){
            if(m.second > cnt){
                cnt = m.second;
                res = m.first;
            }
        }
        return res;
    }
};

## Solution 3

### Runtime : 9 ms , Memory : 7.8 MB 

```c++
class Solution {
public:
    string mostCommonWord(string paragraph, vector<string>& banned) {
        unordered_set<string> ban(banned.begin() , banned.end());
        unordered_map<string,int> mp;
        for(auto & p : paragraph){
            p = isalpha(p) ? tolower(p) : ' ';
        }
        pair<string, int> res("",0);
        istringstream iss(paragraph);
        string w;
        while(iss >> w){
            if(ban.find(w) == ban.end() && ++mp[w] > res.second) {
                res = make_pair(w,mp[w]);
            }
        }
        return res.first;
    }
};
```
