# 125. Valid Palindrome

## Solution

```c++
class Solution {
public:
    
    bool isPalindrome(string s) {
        transform(s.begin() , s.end() , s.begin() , ::tolower);
        string str = "";
        for(int i=0; i<s.size(); i++){
            if( (s[i] >= 'a' && s[i] <= 'z') || (s[i] >= '0' && s[i] <= '9' ) ) {
                str += s[i];
            }
        }
        int n = str.length();
        for(int i=0; i<n/2; i++){
            if(str[i] != str[n-1-i]) return false;
        }
        return true;
    }
};
```
