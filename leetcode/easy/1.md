# 1. Two Sum


## Solution 1

### 시간 복잡도 : O(n^2)

```c++
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        int len = nums.size();
        vector<int> ans;
        bool flag = false;
        for(int i=0; i<len; i++){
            for(int j=i+1; j<len; j++){
                if(nums[i] + nums[j] == target){
                    ans.push_back(i);
                    ans.push_back(j);
                    flag = true;
                    break;
                   
                }
            }
            if(flag) break;
        }
        return ans;
    }
};
```


## Solution 2

### 시간 복잡도 : O(N)


```c++
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int,int> mp;
        vector<int> ans;
        for(int i=0; i<nums.size(); i++) {
            int numTofind = target - nums[i];
            if(mp.find(numTofind) != mp.end()){
                ans.push_back(mp[numTofind]);
                ans.push_back(i);
                break;
            }
            
            mp[nums[i]] = i;
        }
        return ans;
    }
};
```
